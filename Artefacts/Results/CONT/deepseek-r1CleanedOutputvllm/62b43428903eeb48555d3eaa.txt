def formatmany(
    self,
    sql: AnyStr,
    many_params: Union[Iterable[Dict[Union[str, int], Any]], Iterable[Sequence[Any]]],
) -> Tuple[AnyStr, Union[List[Dict[Union[str, int], Any]], List[Sequence[Any]]]]:
    """
    Convert sql using self._converter.convert_many

    Args:
        sql: The SQL string to convert
        many_params: An iterable of dictionaries or sequences to process

    Returns:
        A tuple containing the converted SQL string and the processed parameters
    """
    processed_params = [list(params) if isinstance(params, dict) else list(params) for params in many_params]
    converted_sql, converted_params = self._converter.convert_many(sql, processed_params)
    return converted_sql, converted_params
