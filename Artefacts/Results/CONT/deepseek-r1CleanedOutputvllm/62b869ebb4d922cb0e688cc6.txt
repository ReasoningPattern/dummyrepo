def update_last_applied_manifest_list_from_resp(
    last_applied_manifest, observer_schema, response
):
    """
    Together with :func:``update_last_applied_manifest_dict_from_resp``, this
    function is called recursively to update a partial ``last_applied_manifest``
    from a partial Kubernetes response

    Args:
        last_applied_manifest (list): partial ``last_applied_manifest`` being
            updated
        observer_schema (list): partial ``observer_schema``
        response (list): partial response from the Kubernetes API.

    This function go through all observed fields, and initialized their value in
    last_applied_manifest if they are not yet present
    """
    # Create a copy of the last_applied_manifest to avoid modifying the original
    updated_list = last_applied_manifest.copy()
    
    # Iterate through each item in the response
    for item in response:
        # Check if the item is not already in the updated list
        if item not in updated_list:
            # Append the item to the updated list
            updated_list.append(item)
    
    # Update the last_applied_manifest with the newly updated list
    last_applied_manifest[:] = updated_list
