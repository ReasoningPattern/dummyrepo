def get_config():
    """
    Return a new VersioneerConfig() and set various attribute of it.
    """
    config = VersioneerConfig()
    
    config.git_get_keywords(versionfile_abs=".")
    config.git_versions_from_keywords(
        keywords=git_get_keywords(versionfile_abs=".".),
        tag_prefix="tag-",
        verbose=False
    )
    config.git_pieces_from_vcs(
        tag_prefix="tag-",
        root=".",
        verbose=False
    )
    config.plus_or_dot(pieces=["tag", "dev"])
    config.render_pep440(["tag", "dev"])
    config.render_pep440_branch(["tag", "dev"])
    config.pep440_split_post(ver="20231012")
    config.render_pep440_pre(["tag", "dev"])
    config.render_pep440_post(["tag", "dev"])
    config.render_pep440_post_branch(["tag", "dev"])
    config.render_pep440_old(["tag", "dev"])
    config.render_git_describe(["tag", "dev"])
    config.render_git_describe_long(["tag", "dev"])
    config.get_version()
    config.get_release_version()
    
    return config
