def namesAndDescriptions(self, all=False):
    """
    Returns the name and description of the current class attribute.
    If all=True, returns all names with their descriptions (including None).
    If all=False, returns only names with non-None descriptions.
    """
    names = self.names(all)
    result = []
    for name in names:
        desc = self.getDescriptionFor(name)
        if all or desc is not None:
            result.append((name, desc))
    return result
