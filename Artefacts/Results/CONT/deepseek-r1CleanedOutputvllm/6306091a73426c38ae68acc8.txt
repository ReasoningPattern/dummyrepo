def list_of_file_names(settings_dirs, spec_option):
    """
    Create and return a new IniType complex type via cli.ListOfFileNames()
    """
    from infrared import complex_type

    class FileDirectories(complex_type):
        name = 'FileDirectories'
        cls = os.path.join
        
        def __init__(self, base_dir, pattern=None):
            super().__init__(base_dir, pattern)
            
    return FileDirectories(settings_dirs, spec_option)
