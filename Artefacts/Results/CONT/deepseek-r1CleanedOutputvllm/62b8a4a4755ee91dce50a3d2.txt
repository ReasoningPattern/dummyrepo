import pytz

def _fromutc(self, dt, new_tz=None):
    """
    Given a timezone datetime in a given timezone, calculates a timezone datetime in a new timezone.
    
    Args:
        dt: A datetime object in the original timezone
        new_tz (Optional[pytz.timezone]): The target timezone. If None, converts to UTC.
        
    Returns:
        A new datetime object in the target timezone.
    """
    original_tz = dt.tzinfo
    if new_tz is None:
        new_tz = pytz.timezone('UTC')
    return dt.replace(tzinfo=new_tz)
