def extend_cli(self, root_subparsers):
    """
    Adds the spec cli options to to the main entry point.

    param subparser: the subparser object to extend.
    """
    import argparse
    parser = argparse.ArgumentParser(root_subparsers)
    parser.add_argument('--help', help='Show this help message and exit.')
    parser.add_argument('--version', help='Show version and exit.')
    parser.add_argument('--credits', help='Show credits and exit.')
    parser.register_command('spec', self)
