import argparse
import sys

def parse_arguments(*unparsed_arguments):
    """
    Parses parameters and returns them as dict maps
    """
    parsed_args = argparse.ArgumentParser.parse_args(unparsed_arguments)
    if parsed_args.error:
        print(f"Error: {parsed_args.error}")
        sys.exit(1)
    
    result = {}
    for key, value in vars(parsed_args).items():
        result[key] = value
    return result
