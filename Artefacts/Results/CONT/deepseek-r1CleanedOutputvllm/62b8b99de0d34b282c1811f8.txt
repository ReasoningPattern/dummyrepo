import logging

def _reset_logging(cls):
    """
    Reset the logging system
    """
    # Reset root logger
    logging.getLogger('').handlers = []
    logging.getLogger('').propagating = False
    logging.getLogger('').root = None
    
    # Reset all loggers
    for logger in logging.getloggers():
        logger.handlers = []
        logger.setPropagating(False)
        logger.removeHandler(logging.Handler)
        logger.removeOtherHandler(logging.Handler)
    
    # Reset loggers used by handlers
    for handler in logging.handlers:
        handler.logger.handlers = []
        handler.logger.setPropagating(False)
    
    # Reset loggers used by handlers of handlers
    for handler in logging.handlers:
        for h in handler.logger.handlers:
            h.logger.handlers = []
            h.logger.setPropagating(False)
