import inspect
import subprocess

def subprocess_run_helper(func, *args, timeout, extra_env=None):
    """
    Run a function in a sub-process

    Args:
            func: function, the function to be run. It must be in a module that is importable.
            *args: str. Any additional command line arguments to be passed in
                        the first argument to ``subprocess.run``.
            timeout: timeout for subprocess.run
            extra_env: dict[str, str]. Any additional environment variables to be set for the subprocess.
    Returns:
            CompletedProcess instance
    """
    args_list = list(map(str, (list(inspect.signature(func).parameters.values()) + list(args))))
    return subprocess.run(
        ' '.join(args_list),
        timeout=timeout,
        envvar=extra_env if extra_env is not None else {}
    )
