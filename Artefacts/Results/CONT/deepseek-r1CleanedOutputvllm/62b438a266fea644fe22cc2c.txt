import argparse

def make_parsers():
    """
    Build a top-level parser and its subparsers and return them as a tuple.
    """
    main_parser = argparse.ArgumentParser(description='Top-level parser')
    main_parser.add_subparsers(help action='display help')

    subparser_names = []

    def add_subparser(name, command):
        subparser = argparse.ArgumentParser(name, description=name)
        subparser.add_argument('--' + name, action='store', dest=name)
        subparser.set_defaults(func=command)
        main_parser.add_subparser(subparser)
        subparser_names.append(name)
        return subparser

    # Example subparsers (you would typically create these elsewhere)
    add_subparser('subparser1', lambda: None)
    add_subparser('subparser2', lambda: None)

    return (main_parser,) + tuple(subparser_names)
