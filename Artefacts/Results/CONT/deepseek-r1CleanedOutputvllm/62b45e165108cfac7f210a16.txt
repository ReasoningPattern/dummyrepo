def validate_as_prior_version(self, prior):
    """
    Check that prior is a valid prior version of the current inventory object.
    Both self and prior are instances of InventoryValidator.
    """
    self_version = [int(x) for x in self.spec_version.split('.')]
    prior_version = [int(x) for x in prior.spec_version.split('.')]
    
    for i, (s, p) in enumerate(zip(self_version, prior_version)):
        if p > s:
            self.error(f"Prior version {prior.spec_version} exceeds current version {self.spec_version}")
