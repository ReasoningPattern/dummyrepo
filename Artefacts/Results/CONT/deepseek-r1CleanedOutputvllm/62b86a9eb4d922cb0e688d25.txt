import re

def _get_resource_name_regex():
    """
    Return the regular expressions that are used to validate the name of the Krake resources
    """
    _resource_name_pattern = re.compile(r'^[a-zA-Z0-9_\-].*$', re.IGNORECASE)
    _resource_name_regex = re.compile(r'^[a-zA-Z0-9_.\-].*$', re.IGNORECASE)
    return (_resource_name_pattern, _resource_name_regex)
