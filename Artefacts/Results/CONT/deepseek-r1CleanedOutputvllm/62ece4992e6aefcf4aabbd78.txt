import socket

def is_local(host):
    """
    Checks if the host is the localhost, the localhost include local IP, user name, local domain name, `localhost` and `127.0.0.1`
    
    Args:
        host: The hostname or ip
        
    Returns:
        True if the host is the localhost else False
    """
    if not host:
        return False
    try:
        ip = socket.inet_aton(host)
        return ip == '127.0.0.1'
    except ValueError:
        try:
            resolved = socket.gethostbyname(host)
            return resolved == '127.0.0.1'
        except socket.gaierror:
            return False
