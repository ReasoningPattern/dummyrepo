def append_text_to_file(file_name, text_buffer, encoding, overwrite=False):
    """
    Writes the data in the text buffer to a file
    """
    try:
        with open(file_name, 'a', encoding=encoding) as file:
            if not self.is_file_exist(file_name) and not overwrite:
                file.write('\n')  # Create new file with empty line
            file.write(text_buffer.rstrip('\n'))
            if overwrite or not self.is_file_exist(file_name):
                self.file_to_textbuffer(file_name, encoding).truncate()
        logger.info(f"Successfully wrote text to {file_name}")
        return True
    except Exception as e:
        logger.error(f"Failed to write text to {file_name}: {str(e)}")
        return False
