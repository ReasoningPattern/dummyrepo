def parser_flags(parser):
    """
    Given an argparse.ArgumentParser instance, return its argument flags in a space-separated string.
    Args:
        parser: argparse.ArgumentParser instance

    Returns:
        argument flags in a space-separated string
    """
    flags = [key for key in parser.namespace.__dict__ if key not in ('__weakref__') and parser.namespace.__dict__[key] is not None]
    return ' '.join(flags)
