def get_deprecated_args(self):
    """
    Return a dictionary with options that deprecate others, obtained from self.spec_helper.iterate_option_specs().
    """
    deprecated = {}
    for spec in self.spec_helper.iterate_option_specs():
        if hasattr(spec, 'is_DEPRECATED') and spec.is_DEPRECATED:
            deprecated.update({spec.get_env_option(name): True for name in spec.get_env_option_names()})
    return deprecated
