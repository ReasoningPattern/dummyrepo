def validate_min_max_args(self, args):
    """
    Check if value of the given args is between minimum and maximum values

    Args:
        args (list): List of values to validate

    Raises:
        ValueError: If any value is not within the specified range
    """
    min_val = self.min_val
    max_val = self.max_val

    for value in args:
        if not (min_val <= value <= max_val):
            raise ValueError(f"Value {value} is not within the range {min_val}-{max_val}")
