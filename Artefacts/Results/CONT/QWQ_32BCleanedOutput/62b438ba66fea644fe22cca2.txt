def deep_merge_nodes(nodes):
    merged = {}
    for node in nodes:
        for key, value in node.items():
            if key in merged:
                existing = merged[key]
                if isinstance(existing, dict) and isinstance(value, dict):
                    merged[key] = deep_merge_nodes([existing, value])
                else:
                    merged[key] = value
            else:
                merged[key] = value
    return merged
