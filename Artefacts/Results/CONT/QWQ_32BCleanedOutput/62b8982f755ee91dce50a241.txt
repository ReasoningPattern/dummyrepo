def normalized(self):
    days = self.days
    hours = self.hours
    minutes = self.minutes
    seconds = self.seconds

    days_int = int(days)
    hours += (days - days_int) * 24
    days = days_int

    hours_int = int(hours)
    minutes += (hours - hours_int) * 60
    hours = hours_int

    minutes_int = int(minutes)
    seconds += (minutes - minutes_int) * 60
    minutes = minutes_int

    seconds_int = int(seconds)

    return self.__class__(
        days=days,
        hours=hours,
        minutes=minutes,
        seconds=seconds_int
    )
