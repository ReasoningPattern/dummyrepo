import os

def _group_files_by_xml_filename(source, xmls, files):
    grouped = {xml: [] for xml in xmls}
    for file in files:
        base_file = os.path.splitext(file)[0]
        for xml in xmls:
            base_xml = os.path.splitext(xml)[0]
            if base_file == base_xml:
                grouped[xml].append(file)
                break
    return grouped
