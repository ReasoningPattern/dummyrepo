def parse_subparser_arguments(unparsed_arguments, subparsers):
    parsed_dict = {}
    remaining_args = []
    for subparser_name, subparser in subparsers.items():
        namespace = subparser.parse_args(unparsed_arguments)
        parsed_dict[subparser_name] = namespace
        # Remove the parsed arguments from the unparsed_arguments list
        # This is a simplified approach, assuming that the parsed arguments are the first few in the list
        # In a real implementation, this would require tracking which arguments were parsed
        # For the purpose of this function, we'll assume that the parsed arguments are the first few
        # and the remaining are the rest
        remaining_args = unparsed_arguments
    return (parsed_dict, remaining_args)
