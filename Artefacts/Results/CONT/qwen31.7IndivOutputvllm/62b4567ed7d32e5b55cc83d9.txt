def deep_merge_nodes(nodes):
    result = {}
    for node in nodes:
        if isinstance(node, dict):
            merged = deep_merge_nodes(node.get('children', []))
            for key, value in merged.items():
                if key in result:
                    if not isinstance(value, dict):
                        result[key] = value
                else:
                    result[key] = value
        else:
            key = node
            value = node
            if key in result:
                if not isinstance(value, dict):
                    result[key] = value
            else:
                result[key] = value
    return result
