def deep_merge_nodes(nodes):
    result = {}
    for node in nodes:
        if isinstance(node, dict):
            for key, value in node.items():
                if key in result:
                    if isinstance(value, dict):
                        deep_merge_nodes([value])
                    else:
                        result[key] = value
                else:
                    result[key] = value
        else:
            pass
    return result
