def popitem(self):
    """
    Remove and return the (key, value) pair least frequently used.
    """
    if not self.cache:
        return None
    min_count = min(self.counter.values())
    for key, value in self.cache.items():
        if self.counter[key] == min_count:
            del self.cache[key]
            del self.counter[key]
            return (key, value)
    return None
