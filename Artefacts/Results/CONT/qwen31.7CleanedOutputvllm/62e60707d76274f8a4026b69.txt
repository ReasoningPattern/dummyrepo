def point_type(name, fields, srid_map):
    class Point:
        def __init__(self, *args, **kwargs):
            self.name = name
            self.fields = fields
            self.srid_map = srid_map
            # Initialize attributes based on fields and SRID mapping
            # Example: set SRID from srid_map
            if 'srid' in kwargs:
                self.srid = kwargs['srid']
            else:
                self.srid = srid_map.get('default', 4326)
        def __repr__(self):
            return f"{self.name}({self.fields})"
    return Point
