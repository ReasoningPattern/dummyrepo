def popitem(self):
    min_freq = min(self.freq_groups.keys())
    key = self.freq_groups[min_freq].popleft()
    value = self.cache[key]
    del self.cache[key]
    del self.count[key]
    if not self.freq_groups[min_freq]:
        del self.freq_groups[min_freq]
        if self.freq_groups:
            self.min_freq = min(self.freq_groups.keys())
        else:
            self.min_freq = 0
    return (key, value)
