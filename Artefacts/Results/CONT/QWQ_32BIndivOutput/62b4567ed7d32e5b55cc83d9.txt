def deep_merge_nodes(nodes):
    result = {}
    for node in nodes:
        for key, value in node.items():
            if key in result:
                existing = result[key]
                if isinstance(existing, dict) and isinstance(value, dict):
                    result[key] = deep_merge_nodes([existing, value])
                else:
                    result[key] = value
            else:
                result[key] = value
    return result
