import os

def files_list(path):
    """
    Return the files in given path.
    """
    if not os.path.isdir(path):
        return []
    files = []
    for filename in os.listdir(path):
        full_path = os.path.join(path, filename)
        if os.path.isfile(full_path):
            files.append(filename)
    return files
