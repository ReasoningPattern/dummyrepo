import socket
import ipaddress

def is_local(host):
    if host in ("localhost", "127.0.0.1"):
        return True

    try:
        ip = ipaddress.ip_address(host)
        if ip.is_loopback:
            return True
    except ValueError:
        pass

    try:
        local_hostname = socket.gethostname()
        local_fqdn = socket.getfqdn()
        if host == local_hostname or host == local_fqdn:
            return True

        if '.' in local_fqdn:
            domain = '.'.join(local_fqdn.split('.')[1:])
            if host == domain:
                return True
    except:
        pass

    return False
