

def render(pieces, style):
    if style == 'csv':
        return ','.join(map(str, pieces))
    elif style == 'html_ul':
        return '<ul>' + ''.join(f'<li>{p}</li>' for p in pieces) + '</ul>'
    elif style == 'markdown':
        return '\n'.join(f'* {p}' for p in pieces)
    elif style == 'plain':
        return ' '.join(map(str, pieces))
    else:
        raise ValueError(f"Unsupported style: {style}")
