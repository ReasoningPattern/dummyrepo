import os
import shutil

def normalize_cmd(cmd: tuple[str, ...]) -> tuple[str, ...]:
    if not cmd:
        return cmd
    exe_name = cmd[0]
    if os.path.isabs(exe_name):
        return cmd
    full_path = shutil.which(exe_name)
    if full_path is None:
        return cmd
    return (full_path,) + cmd[1:]
