def get_nested_custom_and_control_args(self, args):
    control_args = {}
    nested_args = {}
    for arg in args:
        parts = arg.split('=', 1)
        key = parts[0]
        value = parts[1] if len(parts) > 1 else ''
        if key.startswith('ctrl_'):
            control_args[key] = value
        else:
            nested_args[key] = value
    return control_args, nested_args
