def format(
		self,
		sql: AnyStr,
		params: Union[Dict[Union[str, int], Any], Sequence[Any]],
	) -> Tuple[AnyStr, Union[Dict[Union[str, int], Any], Sequence[Any]]]:
		"""
		Convert sql using self._converter.convert
		"""
		formatted_sql = self._converter.convert(sql, params)
