def get_silent_args(self, args):
    """
    list of silenced argument

    :param args: The received arguments.
    :return: list, silenced argument names
    """
    silenced = []
    for arg in self._iterate_received_arguments(self, args):
        if getattr(arg, 'silent', False):
            silenced.append(arg.name)
    return silenced
