def next_version(version):
    """
    Given next version identifier following existing pattern
    """
    parts = list(map(int, version.split('.')))
    carry = 1
    for i in range(len(parts)-1, -1, -1):
        new_val = parts[i] + carry
        if new_val == 10:
            parts[i] = 0
            carry = 1
        else:
            parts[i] = new_val
            carry = 0
            break
    if carry == 1:
        parts.insert(0, 1)
    return '.'.join(map(str, parts))
