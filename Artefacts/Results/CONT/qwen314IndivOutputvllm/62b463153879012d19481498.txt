def files_list(path):
    """
    Return the files in given path.
    """
    try:
        entries = os.listdir(path)
        files = [f for f in entries if os.path.isfile(os.path.join(path, f))]
        return files
    except Exception as e:
        logging.error(f"Error listing files in {path}: {e}")
        return []
