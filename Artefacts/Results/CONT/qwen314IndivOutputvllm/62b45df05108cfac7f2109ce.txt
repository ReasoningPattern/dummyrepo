def validate(self, path):
    """
    Returns True if OCFL object at path or pyfs root, False otherwise.
    """
    if not self.validate_object_root(path):
        return False
    if not self.validate_inventory(self.content_directory, "root", self.spec_version):
        return False
    if not self.validate_inventory_digest_match(self.content_directory / "inventory.json", self.inventory_digest_files):
        return False
    if self.registered_extensions:
        if not self.validate_extensions_dir(self.content_directory):
            return False
    if not self.validate_version_inventories(self.version_dirs):
        return False
    if not self.validate_content(self.inventory, self.version_dirs, {}, {}):
        return False
    if not self.check_additional_digests(path, self.read_inventory_digest(self.inventory_digest_files), self.additional_digests, 1):
        return False
    return True
