def initialize(self):
    """
    Create and initialize a new OCFL storage root.
    """
    self.open_root_fs(create=True)
    self.check_root_structure()
    self.parse_layout_file()
    self.validate_hierarchy(validate_objects=True, check_digests=self.lax_digests, show_warnings=True)
    self.root_fs.makedirs(self.layout_file, exist_ok=True)
    self.root_fs.makedirs(self.spec_file, exist_ok=True)
    self.root_fs.makedirs(self.registered_extensions, exist_ok=True)
    self.log.info("OCFL storage root initialized at %s", self.root)
