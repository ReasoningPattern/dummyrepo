def get_versions():
    """
    Obtains the version information. If the version information cannot be obtained, the default value is returned.
    """
    try:
        # Attempt to get version from parent directory
        version = versions_from_parentdir(parentdir_prefix="v", root=os.path.dirname(__file__), verbose=False)
        if version:
            return version
    except Exception:
        pass

    try:
        # Attempt to get version from Git
        version = git_versions_from_keywords(git_get_keywords(os.path.join(os.path.dirname(__file__), ".version")), tag_prefix="v", verbose=False)
        if version:
            return version
    except Exception:
        pass

    # Return default version if all else fails
    return "0.0.0"
