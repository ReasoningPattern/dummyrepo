import functools

def cached(cache, key=functools.hashkey, lock=None):
    """
    Returns a decorator function that saves the results in the cache
    """
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            k = key(args, kwargs)
            if lock is not None:
                with lock:
                    if k in cache:
                        return cache[k]
                    result = func(*args, **kwargs)
                    cache[k] = result
                    return result
            else:
                if k in cache:
                    return cache[k]
                result = func(*args, **kwargs)
                cache[k] = result
                return result
        return wrapper
    return decorator
