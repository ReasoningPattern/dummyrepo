def initialize(self):
    """
    Create and initialize a new OCFL storage root.
    """
    self.open_root_fs(create=True)
    self.check_root_structure()
    self.parse_layout_file()
    self.validate_extensions_dir()
    self.disposition = self._dispositor()
    self.layout_file = self.parse_layout_file()
    self.registered_extensions = self.validate_extensions_dir()
    self.description = self.layout_file.get('description', '')
    self.good_objects = 0
    self.num_objects = 0
    self.num_traversal_errors = 0
    self.lax_digests = False
    self.log.info("OCFL storage root initialized successfully")
