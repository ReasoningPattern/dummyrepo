def absorb(self, args):
    """
    Given an `args` sequence of expressions, return a new list of expression applying absorption.
    """
    new_args = []
    for expr in args:
        absorbed = False
        for existing in new_args:
            if self.subtract(existing, expr):
                absorbed = True
                break
        if not absorbed:
            new_args.append(expr)
    return new_args
