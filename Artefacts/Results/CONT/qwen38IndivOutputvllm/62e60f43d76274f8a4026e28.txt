import datetime
import pytz

def hydrate_time(nanoseconds, tz=None):
    """
    Convert nanoseconds to a time in fixed format.
    """
    seconds = nanoseconds / 1_000_000_000.0
    if tz is None:
        tz = pytz.UTC
    dt = datetime.datetime.fromtimestamp(seconds, tz=tz)
    return dt
