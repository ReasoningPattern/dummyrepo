def difference(d1, d2, level=-1):
    """
    Return a dictionary with items from d1 not contained in d2.
    """
    result = {}
    for key in d1:
        if key not in d2:
            result[key] = d1[key]
        else:
            if isinstance(d1[key], dict) and isinstance(d2[key], dict):
                if level == -1:
                    sub_level = -1
                else:
                    sub_level = level - 1
                sub_diff = difference(d1[key], d2[key], sub_level)
                if sub_diff:
                    result[key] = sub_diff
            else:
                if d1[key] != d2[key]:
                    result[key] = d1[key]
    return result
