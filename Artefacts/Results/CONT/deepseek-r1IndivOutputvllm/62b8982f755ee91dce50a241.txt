def normalized(self):
    """
    Normalize all units of time to integers.
    """
    total_seconds = (self.years * 365.25 * 24 * 60 * 60 +
                     self.months * 30.44 * 24 * 60 * 60 +
                     self.days * 24 * 60 * 60 +
                     self.hours * 60 * 60 +
                     self.minutes * 60 +
                     self.seconds)
    return math.floor(total_seconds)
