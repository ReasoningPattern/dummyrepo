def parser_flags(parser):
    flags = []
    for key, value in parser.collect_args().__dict__.items():
        if value is not None:
            if isinstance(value, bool):
                flags.append(f"{key}=true" if value else f"{key}=false")
            else:
                flags.append(f"{key}={str(value)}")
    return ' '.join(flags)
