def fill(self, coord=None, weight=1):
    """
    Fill histogram at *coord* with the given *weight*.

    Coordinates outside the histogram edges are ignored.
    """
    if coord is None:
        return self
        
    self._validate_coord(coord)
    bin_index = self._get_bin_index(coord)
    self.bins[bin_index] = [w + weight for w in self.bins[bin_index]]
    return self
