def validate_choices_args(self, args):
    """
    Check if value of the given args is one of the available choices.
    
    :param args: The received arguments.
    :return: True if the argument is a valid choice, False otherwise.
    """
    if not hasattr(self, 'choices'):
        raise AttributeError("self.choices must be defined")
        
    possible_values = self.choices.keys()
    value = str(args[0])
    
    return value in possible_values

# Example usage:
# valid_value = validate_choices_args(['a'])  # Returns True if 'a' is a valid choice
# invalid_value = validate_choices_args(['c'])  # Returns False if 'c' is not a valid choice
