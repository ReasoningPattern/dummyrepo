def strip_root(path, root):
    """
    Remove root from path. If fails, throw exception

    Args:
        path (str): The full path to strip the root from
        root (str): The root directory to remove from the path

    Returns:
        str: The path without the root directory

    Raises:
        ValueError: If the path does not start with the root directory
    """
    if not path.startswith(root):
        raise ValueError("Path does not start with the root directory")
    return path[len(root):]
