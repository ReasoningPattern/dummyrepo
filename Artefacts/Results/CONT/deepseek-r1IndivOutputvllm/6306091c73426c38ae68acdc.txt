import os
import jsonschema
from six import string_types
import infrared

class SpecValidator(string_types):
    def __init__(self, spec_schema):
        super(SpecValidator, self).__init__()
        self.spec_schema = spec_schema

    def validate_from_content(self, cls, spec_content=None):
        if spec_content is None:
            raise InvalidSpecificationError("No spec content provided")
        
        try:
            schema = jsonschema.load(self.spec_schema)
            validator = jsonschema.Validator(schema)
            validator.check(spec_content)
        except jsonschema.ValidationError as e:
            raise InvalidSpecificationError(f"Missing required fields: {e detail}")

        data = json.loads(spec_content)
        return data

class InvalidSpecificationError(ValueError):
    pass
