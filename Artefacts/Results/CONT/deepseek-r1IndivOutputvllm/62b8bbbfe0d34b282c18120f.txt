import codecs
import os
import logging

logger = logging.getLogger(__name__)

def file_to_textbuffer(file_name, encoding):
    """
    Load a file toward a text buffer
    
    Args:
        file_name (str): Name of the file to load
        encoding (str): Encoding to use when reading the file
    
    Returns:
        str: Content of the file as a text buffer
    """
    try:
        with codecs.open(file_name, 'r', encoding=encoding) as f:
            text_buffer = f.read()
        logger.info(f"Successfully loaded file: {file_name}")
        return text_buffer
    except Exception as e:
        logger.error(f"Failed to load file: {str(e)}")
        raise
