def _normalizeargs(sequence, output=None):
    """
    Normalize declaration arguments

    Normalization arguments might contain Declarations, tuples, or single
    interfaces. The function expands any complex argument into its constituent
    parts, collecting all individual arguments.
    """
    result = []
    for arg in sequence:
        if isinstance(arg, Declarments):
            result.extend(arg.interfaces)
        elif isinstance(arg, tuple):
            result.extend(arg)
        else:
            result.append(arg)
    if output is not None:
        output.extend(result)
    return result
