import inspect

def generate_default_observer_schema_dict(manifest_dict, first_level=False):
    """
    Generates a schema dictionary based on the value types in the input manifest dictionary.
    """
    result = {}
    for key, value in manifest_dict.items():
        if isinstance(value, dict):
            if first_level:
                result[key] = {'type': 'dict'}
            else:
                sub_schema = {}
                for sub_key, sub_value in value.items():
                    sub_schema[sub_key] = inspect.gettype(sub_value)
                result[key] = sub_schema
        elif isinstance(value, list):
            if first_level:
                result[key] = {'type': 'list'}
            else:
                sub_schema = {}
                for i, item in enumerate(value):
                    if isinstance(item, dict):
                        item_schema = {}
                        for sub_key, sub_value in item.items():
                            item_schema[sub_key] = inspect.gettype(sub_value)
                        sub_schema[i] = item_schema
                    else:
                        sub_schema[i] = inspect.gettype(item)
                result[key] = sub_schema
        else:
            result[key] = value
    return result
