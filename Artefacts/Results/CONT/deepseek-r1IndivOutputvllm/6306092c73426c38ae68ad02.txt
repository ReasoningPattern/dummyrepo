import warnings

def validate_arg_deprecation(self, cli_args, answer_file_args):
    """
    Validates and prints the deprecated arguments.

   :param cli_args: the dict of arguments from cli
   :param answer_file_args:  the dict of arguments from files
    """
    # Get the list of deprecated arguments
    deprecated_args = self.get_deprecated_args()
    
    # Check for deprecated arguments in both cli_args and answer_file_args
    for arg_name, arg_type in deprecated_args.items():
        if arg_name in cli_args and isinstance(cli_args[arg_name], arg_type):
            warnings.warn(f"Deprecated argument '{arg_name}' provided. Use new argument instead.", UserWarning)
        if arg_name in answer_file_args and isinstance(answer_file_args[arg_name], arg_type):
            warnings.warn(f"Deprecated argument '{arg_name}' provided in answer file. Use new argument instead.", UserWarning)
