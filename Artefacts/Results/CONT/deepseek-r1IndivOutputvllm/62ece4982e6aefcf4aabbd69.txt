def _replace_register(flow_params, register_number, register_value):
    if register_number < 0 or register_number >= len(flow_params) or register_value < 0 or register_value >= len(flow_params):
        raise IndexError("register number or value out of bounds")
    if register_number == register_value:
        raise ValueError("register number and value cannot be the same")
    value = flow_params.pop(register_value)
    flow_params[register_number] = value
