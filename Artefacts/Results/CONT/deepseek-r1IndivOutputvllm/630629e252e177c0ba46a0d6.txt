

```python
import typing
import logging
import urllib
import xml
import json
import xrd
import federation
import lxml

def retrieve_diaspora_host_meta(host: str) -> xrd.XRD:
    """
    Retrieve a remote Diaspora host-meta document.
    
    :arg host: Host to retrieve from
    :returns: ``xrd.XRD`` instance
    """
    logging.debug(f"Fetching host meta data from {host}")
    
    request = urllib.request.Request(
        f"{host}/_rest/api/2.0/metadata/remote",
        headers={'XRD-VERSION': '2023-10-01', 'Host': host}
    )
    
    try:
        response = urllib.request.urlopen(request)
        if not response.getcode() == 200:
            raise urllib.error.HTTPError(
                reason=str(response.getcode()),
                body=xml.dom.minidom.parseString(response.read()).read()
            )
        xm = xml.parseString(response.read())
        return xrd.XRD(xm)
    except urllib.error.HTTPError as err:
        logging.error(f"HTTP Error: {err.reason}")
        raise
```
