def pop_u16(self):
    """
    Remove the last two elements in self.data and return
    """
    if len(self.data) < 2:
        raise IndexError("pop from an empty sequence")
    p16 = self.data[-1]
    p16_prev = self.data[-2]
    self.data = self.data[:-2]
    p16_int = struct.unpack('I', bytes.fromhex(p16))[0]
    p16_prev_int = struct.unpack('I', bytes.fromhex(p16_prev))[0]
    return (p16_int, p16_prev_int)
