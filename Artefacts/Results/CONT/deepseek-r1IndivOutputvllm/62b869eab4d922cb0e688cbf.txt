def generate_default_observer_schema(app):
    """
    Generate the default observer schema for each Kubernetes resource present in ``spec.manifest`` for which a custom observer schema hasn't been specified.
    """
    from typing import Dict, Any

    observer_schemas: Dict[str, Any] = {}
    
    for resource in app['spec'].get('resources', []):
        if 'observer_schema' not in resource['spec']:
            default_schema = {
                'kind': 'ServerSide',
                'name': resource['metadata'].get('name'),
                'namespace': resource['metadata'].get('namespace'),
                'api_version': 'v1',
                'server_side': True,
                'observer_schema': 'default_observer_schema',
                'metadata': {
                    'name': resource['metadata'].get('name'),
                    'labels': resource['metadata'].get('labels', [])
                }
            }
            observer_schemas[resource['metadata'].get('name')] = default_schema
    
    return observer_schemas
