def validate_choices_args(self, args):
    """
    Check if value of the given args is one of the available choices. \n:param args: The received arguments.
    """
    for arg, value in args.items():
        if arg in self.choices and value not in self.choices[arg]:
            raise ValueError(f"Invalid value for {arg}: {value}. Must be one of {self.choices[arg]}")
