def xml_children_as_dict(node):
    """
    Turn the children of node <xml> into a dict, keyed by tag name.
    """
    result = {}
    for child in node:
        tag = child.tag
        if tag in result:
            if not isinstance(result[tag], list):
                result[tag] = [result[tag]]
            result[tag].append(child)
        else:
            result[tag] = child
    return result
