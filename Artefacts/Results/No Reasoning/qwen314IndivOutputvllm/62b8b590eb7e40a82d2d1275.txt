def _legacy_mergeOrderings(orderings):
    """
    Combine multiple lists in the sequence of occurrence into a list with no duplicate elements.
    """
    result = []
    seen = set()
    for lst in orderings:
        for item in lst:
            if item not in seen:
                seen.add(item)
                result.append(item)
    return result
