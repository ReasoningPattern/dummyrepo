def minimalBases(classes):
	"""
	Returns all classes without subclasses as a list.
	"""
	subclasses = set()
	for cls in classes:
		for subclass in classes:
			if issubclass(subclass, cls) and subclass != cls:
				subclasses.add(cls)
				break
	return [cls for cls in classes if cls not in subclasses]
