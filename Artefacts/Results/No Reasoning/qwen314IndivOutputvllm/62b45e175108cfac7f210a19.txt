def validate_fixity(self, fixity, manifest_files):
    """
    Validate fixity block in inventory. Check the structure of the fixity block and makes sure that only files listed in the manifest are referenced. Return error() in the class.
    """
    if not isinstance(fixity, list):
        self.error("Fixity block must be a list")
        return

    for entry in fixity:
        if not isinstance(entry, dict) or "file" not in entry or "checksum" not in entry:
            self.error("Invalid fixity entry structure")
            return

        file_path = entry["file"]
        if file_path not in manifest_files:
            self.error(f"File referenced in fixity block not found in manifest: {file_path}")
            return
