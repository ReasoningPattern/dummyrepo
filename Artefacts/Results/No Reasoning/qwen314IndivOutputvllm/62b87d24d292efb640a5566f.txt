def render(pieces, style):
    """
    Input pieces and a style, render the pieces to the corresponding style.
    """
    if style == 'html':
        return '<div>' + '</div><div>'.join(pieces) + '</div>'
    elif style == 'text':
        return '\n'.join(pieces)
    else:
        raise ValueError(f"Unsupported style: {style}")
