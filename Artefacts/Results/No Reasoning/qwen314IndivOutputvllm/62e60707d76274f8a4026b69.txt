def point_type(name, fields, srid_map):
    """
    Dynamically Generating Point Class
    """
    class Point:
        def __init__(self, **kwargs):
            for field in fields:
                setattr(self, field, kwargs.get(field))
            self.srid = kwargs.get('srid')
            self.srid_map = srid_map

        def __repr__(self):
            return f"{name}({', '.join([f'{field}={getattr(self, field)}' for field in fields])}, srid={self.srid})"

        def to_wkt(self):
            srid = self.srid_map.get(self.srid, 4326)
            coords = ', '.join([str(getattr(self, field)) for field in fields])
            return f"POINT ({coords}) SRID={srid}"

    return Point
