def hydrate_time(nanoseconds, tz=None):
    """
    Convert nanoseconds to a time in fixed format.
    """
    from datetime import datetime, timezone

    if tz is None:
        tz = timezone.utc

    seconds = nanoseconds // 1_000_000_000
    nanoseconds_remainder = nanoseconds % 1_000_000_000

    dt = datetime(1, 1, 1, tzinfo=tz)
    dt += dt.resolution * seconds
    dt += dt.resolution * (nanoseconds_remainder / 1_000_000_000)

    return dt.time()
