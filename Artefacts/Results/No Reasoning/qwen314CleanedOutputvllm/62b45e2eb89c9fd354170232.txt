def next_version(version):
    """
    Given next version identifier following existing pattern
    """
    parts = list(map(int, version.split('.')))
    if len(parts) == 1:
        return str(parts[0] + 1)
    elif len(parts) == 2:
        return f"{parts[0]}.{parts[1] + 1}"
    else:
        return f"{parts[0]}.{parts[1]}.{parts[2] + 1}"
