def verifyClass(iface, candidate, tentative=False):
    """
    Verify that the *candidate* might correctly provide *iface*.
    """
    if not tentative:
        if not isinstance(candidate, iface):
            raise TypeError(f"Expected {iface}, got {type(candidate)}")
    else:
        if not issubclass(candidate, iface):
            raise TypeError(f"Expected subclass of {iface}, got {candidate}")
