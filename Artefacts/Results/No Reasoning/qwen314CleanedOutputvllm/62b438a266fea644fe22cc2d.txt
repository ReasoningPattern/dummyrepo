def parse_subparser_arguments(unparsed_arguments, subparsers):
    """
    Given a sequence of arguments and a dict from subparser name to argparse.ArgumentParser
    instance, give each requested action's subparser a shot at parsing all arguments. This allows
    common arguments like "--repository" to be shared across multiple subparsers.

    Return the result as a tuple of (a dict mapping from subparser name to a parsed namespace of
    arguments, a list of remaining arguments not claimed by any subparser).
    """
    parsed = {}
    remaining_args = list(unparsed_arguments)
    for name, parser in subparsers.items():
        try:
            # Make a copy of the remaining arguments to avoid modifying the list during iteration
            args_copy = list(remaining_args)
            # Parse the arguments with the current subparser
            parsed_args = parser.parse_args(args_copy)
            # Update the remaining arguments by removing the ones that were parsed
            remaining_args = [arg for arg in remaining_args if arg not in args_copy or arg in vars(parsed_args)]
            parsed[name] = parsed_args
        except SystemExit:
            # If parsing fails, skip this subparser
            continue
    return parsed, remaining_args
