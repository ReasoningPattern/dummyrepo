def get_config():
    """
    Return a new VersioneerConfig() and set various attribute of it.
    """
    from versioneer import VersioneerConfig
    config = VersioneerConfig()
    config.versionfile_source = "VERSION"
    config.versionfile_build = "version.txt"
    config.tag_prefix = ""
    config.parentdir_prefix = "v"
    return config
