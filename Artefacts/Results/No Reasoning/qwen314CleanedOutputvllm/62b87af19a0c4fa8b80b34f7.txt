def difference(d1, d2, level=-1):
	"""
	Return a dictionary with items from d1 not contained in d2.
	"""
	result = {}
	for key in d1:
		if key not in d2:
			result[key] = d1[key]
		elif level > 0 and isinstance(d1[key], dict) and isinstance(d2[key], dict):
			result[key] = difference(d1[key], d2[key], level - 1)
	return result
