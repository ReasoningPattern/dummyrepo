def is_fill_request_seq(seq):
	"""
	Check whether seq can be converted to FillRequestSeq and bool is returned.
	"""
	try:
		# Check if the sequence is a list of dictionaries with specific keys
		if not isinstance(seq, list):
			return False
		for item in seq:
			if not isinstance(item, dict):
				return False
			if 'symbol' not in item or 'quantity' not in item or 'price' not in item:
				return False
		return True
	except:
		return False
