def validate_min_max_args(self, args):
	"""
	Check if value of the given args is between minimum and maximum values
	"""
	if not args:
		raise ValueError("args cannot be empty")
	if len(args) != 2:
		raise ValueError("args must have exactly two elements")
	val1, val2 = args
	if not (isinstance(val1, (int, float)) and isinstance(val2, (int, float))):
		raise ValueError("both args must be numbers")
	if val1 > val2:
		raise ValueError("minimum value must be less than or equal to maximum value")
	return True
