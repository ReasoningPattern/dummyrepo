def _group_files_by_xml_filename(source, xmls, files):
	"""
	Groups files by xmls and returns data in dict format.
	"""
	grouped_files = {}

	for file in files:
		xml_name = file.split('.')[0]  # Assuming xml files have .xml extension
		if xml_name in grouped_files:
			grouped_files[xml_name].append(file)
		else:
			grouped_files[xml_name] = [file]

	return grouped_files
