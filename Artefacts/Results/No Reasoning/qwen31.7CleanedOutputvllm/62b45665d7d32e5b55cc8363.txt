import argparse

def make_parsers():
    """
    Build a parser and its subparsers and return them as a tuple.
    """
    parser = argparse.ArgumentParser(description="Example parsers")
    subparsers = argparse.ArgumentParser(add_help=False)
    parser.add_subparsers(action='store_true', help='Subparsers')
    return parser, subparsers
