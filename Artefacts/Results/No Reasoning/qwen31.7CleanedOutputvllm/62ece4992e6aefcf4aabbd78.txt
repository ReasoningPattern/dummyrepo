def is_local(host):
    """
    Checks if the host is the localhost,
    the localhost include local IP, user name, local domain name, `localhost` and `127.0.0.1`

    Args:
        host: The hostname or ip

    Returns:
        True if the host is the localhost else False
    """
    import socket
    try:
        # Check if the host is localhost
        if host == 'localhost':
            return True
        # Check if the host is 127.0.0.1
        if host == '127.0.0.1':
            return True
        # Check if the host is the local loopback address
        if socket.inet_pton(socket.AF_INET, host) in socket.inet_pton(socket.AF_INET, '127.0.0.1'):
            return True
        # Check if the host is the local domain name (e.g., localhost.localdomain)
        if socket.inet_pton(socket.AF_INET6, host) in socket.inet_pton(socket.AF_INET6, '::1'):
            return True
        return False
    except:
        return False
