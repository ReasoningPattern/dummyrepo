import logging
from logging import getLogger, basicConfig, FileHandler, DEBUG

def build_app_logger(name='app', logfile='app.log', debug=True):
	"""
	General purpose application logger. Useful mainly for debugging
	Args:
		name: The name of logger
		logfile: The log file need to save
		debug: is need to debug

	Returns:
		Returns an instantiated logger object
	"""
	basicConfig(filename=logfile, level=DEBUG)
	logger = getLogger(name)
	handler = FileHandler(logfile)
	logger.addHandler(handler)
	if debug:
		logger.setLevel(DEBUG)
	return logger
