def size_to_bytes(size: str) -> int:
	"""
	Convert human readable file size to bytes

    Args:
            size: str, a string representing a human readable file size (eg: '500K')
    Returns:
            int: file size in bytes
	"""
	# Split the size string into parts
	parts = size.split()
	
	# Get the numeric part and the unit
	numeric_part = float(parts[0])
	unit = parts[1]
	
	# Define the conversion factors
	conversion_factors = {
		'K': 1024,
		'M': 1024 ** 2,
		'G': 1024 ** 3,
		'T': 1024 ** 4,
		'P': 1024 ** 5,
		'EB': 1024 ** 6,
		'ZB': 1024 ** 7,
		'YB': 1024 ** 8,
	}
	
	# Get the conversion factor
	conversion_factor = conversion_factors.get(unit, 1)
	
	# Calculate the bytes
	return int(numeric_part * conversion_factor)
