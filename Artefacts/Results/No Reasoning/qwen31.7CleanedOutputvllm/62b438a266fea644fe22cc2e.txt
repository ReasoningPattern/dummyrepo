import argparse

def parse_arguments(*unparsed_arguments):
    parser = argparse.ArgumentParser()
    subparsers = parser.add_subparsers(required=True)
    
    # Process each argument
    for arg in unparsed_arguments:
        # This is a simplified example; in a real scenario, you would parse each argument
        # and add it to the subparsers
        subparser = subparsers.add_parser(arg)
        subparser.set_defaults(func=arg)
    
    # Parse the arguments
    args = parser.parse_args(unparsed_arguments)
    
    # Return the parsed arguments as a dict
    return {
        "global": args,
        "subparsers": subparsers
    }
