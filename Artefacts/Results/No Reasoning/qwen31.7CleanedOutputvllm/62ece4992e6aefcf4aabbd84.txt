def is_ipv4(target):
	"""
	Test if IPv4 address or not.

	Returns: Boolean, True if target is IPv4 address, else False.
	"""
	try:
		# Attempt to parse the target as an IPv4 address
		ip = tuple(map(int, target.split('.')))
		if len(ip) != 4:
			return False
		for part in ip:
			if part < 0 or part > 255:
				return False
		return True
	except ValueError:
		return False
