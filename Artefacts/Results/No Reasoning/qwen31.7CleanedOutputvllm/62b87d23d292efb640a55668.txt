def get_config():
    """
    Return a new VersioneerConfig() and set various attribute of it.
    """
    from versioneer import VersioneerConfig
    config = VersioneerConfig()
    config.version_base = '1'
    config.version_files = ['__init__.py']
    config.version_regexp = r'__version__ = \'([0-9]+)\'"
    config.version_template = 'v{version}'
    config.dist_tag = 'development'
    config.python_version = '3.8'
    config.local_version = True
    config.show_version_in_docs = True
    return config
