def check_digests_present_and_used(self, manifest_files, digests_used):
	"""
	Check all digests in manifest that are needed are present and used. Return error() in the class.
	"""
	for digest in digests_used:
		found = False
		for manifest_file in manifest_files:
			if digest in manifest_file:
				found = True
				break
		if not found:
			self.error(f"Digest {digest} is used but not present in any manifest file.")
