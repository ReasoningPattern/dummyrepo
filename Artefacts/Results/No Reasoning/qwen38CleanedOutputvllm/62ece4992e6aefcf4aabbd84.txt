def is_ipv4(target):
    """
    Test if IPv4 address or not.

   Returns: Boolean, True if target is IPv4 address, else False.
    """
    try:
        parts = target.split('.')
        if len(parts) != 4:
            return False
        for part in parts:
            if not part.isdigit():
                return False
            num = int(part)
            if num < 0 or num > 255:
                return False
        return True
    except:
        return False
