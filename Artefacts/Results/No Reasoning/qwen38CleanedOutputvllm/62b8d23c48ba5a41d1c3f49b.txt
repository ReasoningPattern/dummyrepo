def popitem(self):
    """
    Remove and return the (key, value) pair most recently used.
    """
    if not self:
        raise KeyError("popitem() from empty dictionary")
    key, value = self._most_recent_key, self._most_recent_value
    del self[key]
    self._most_recent_key, self._most_recent_value = None, None
    return (key, value)
