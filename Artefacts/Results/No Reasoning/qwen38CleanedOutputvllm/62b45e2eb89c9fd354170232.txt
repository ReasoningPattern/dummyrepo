def next_version(version):
    """
    Given next version identifier following existing pattern
    """
    parts = version.split('.')
    if not parts:
        return "0.0.1"
    if len(parts) == 1:
        return f"0.0.{int(parts[0]) + 1}"
    if len(parts) == 2:
        return f"0.{int(parts[1]) + 1}.0"
    if len(parts) == 3:
        return f"{int(parts[0]) + 1}.0.0"
    return version
