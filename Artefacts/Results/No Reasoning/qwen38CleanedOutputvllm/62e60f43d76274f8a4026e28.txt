def hydrate_time(nanoseconds, tz=None):
    """
    Convert nanoseconds to a time in fixed format.
    """
    import datetime
    if tz is None:
        dt = datetime.datetime(1970, 1, 1) + datetime.timedelta(nanoseconds=nanoseconds)
    else:
        dt = datetime.datetime(1970, 1, 1, tzinfo=tz) + datetime.timedelta(nanoseconds=nanoseconds)
    return dt.strftime('%Y-%m-%d %H:%M:%S.%f')
