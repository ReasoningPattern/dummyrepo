def absorb(self, args):
    """
	Given an `args` sequence of expressions, return a new list of expression applying absorption.
	"""
    result = []
    i = 0
    while i < len(args):
        if i + 1 < len(args) and self._is_absorbable(args[i], args[i + 1]):
            result.append(self._apply_absorption(args[i], args[i + 1]))
            i += 2
        else:
            result.append(args[i])
            i += 1
    return result
