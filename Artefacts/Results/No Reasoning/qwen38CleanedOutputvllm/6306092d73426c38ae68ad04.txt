def get_parser_option_specs(self, command_name):
	"""
	Gets all the options for the specified command

	:param command_name: the command name (main, virsh, ospd, etc...)
	:return: the list of all command options
	"""
	options = []
	if command_name == 'main':
		options = [
			{'option': '--help', 'action': 'help', 'help': 'Show help message and exit'},
			{'option': '--version', 'action': 'version', 'help': 'Show program version and exit'}
		]
	elif command_name == 'virsh':
		options = [
			{'option': '--connect', 'dest': 'connect', 'help': 'Connect to hypervisor'},
			{'option': '--help', 'action': 'help', 'help': 'Show help message and exit'}
		]
	elif command_name == 'ospd':
		options = [
			{'option': '--config', 'dest': 'config', 'help': 'Specify configuration file'},
			{'option': '--debug', 'action': 'store_true', 'help': 'Enable debug mode'}
		]
	return options
