def dehydrate_point(value):
    """
    The structure class is generated based on the value length.
    """
    if len(value) == 1:
        return {'type': 'point', 'coordinates': [value[0]]}
    elif len(value) == 2:
        return {'type': 'point', 'coordinates': value}
    else:
        raise ValueError("Value must have exactly 1 or 2 elements for a point.")
