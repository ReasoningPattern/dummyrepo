def _include_groups(self, parser_dict):
    """
    Resolves the include dict directive in the spec files.
    """
    if 'include' in parser_dict:
        include_paths = parser_dict['include']
        if isinstance(include_paths, str):
            include_paths = [include_paths]
        for path in include_paths:
            try:
                with open(path, 'r') as f:
                    included_content = yaml.safe_load(f)
                for key, value in included_content.items():
                    if key in parser_dict:
                        parser_dict[key].extend(value)
                    else:
                        parser_dict[key] = value
            except FileNotFoundError:
                raise ValueError(f"Include file not found: {path}")
            except yaml.YAMLError as e:
                raise ValueError(f"Error parsing include file {path}: {e}")
