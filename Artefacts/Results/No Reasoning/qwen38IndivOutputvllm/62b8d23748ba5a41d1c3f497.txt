def popitem(self):
    """
	Remove and return the `(key, value)` pair least frequently used.
	"""
    if not self:
        raise KeyError("popitem() from empty LRU cache")
    least_freq_key = next(iter(self._freq))
    key, value = self._cache[least_freq_key]
    del self._cache[least_freq_key]
    self._freq[least_freq_key] -= 1
    if self._freq[least_freq_key] == 0:
        del self._freq[least_freq_key]
    return (key, value)
