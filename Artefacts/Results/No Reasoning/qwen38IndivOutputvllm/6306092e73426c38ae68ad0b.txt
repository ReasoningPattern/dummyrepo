def validate_min_max_args(self, args):
    """
    Check if value of the given args is between minimum and maximum values
    """
    for arg in args:
        if not (self.min_value <= arg <= self.max_value):
            raise ValueError(f"Argument {arg} is out of bounds. Must be between {self.min_value} and {self.max_value}.")
