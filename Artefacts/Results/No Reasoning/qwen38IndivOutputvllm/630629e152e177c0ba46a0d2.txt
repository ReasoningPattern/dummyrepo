import requests
from lxml import etree

def retrieve_and_parse_diaspora_webfinger(handle):
    """
	Retrieve a and parse a remote Diaspora webfinger document.

	:arg handle: Remote handle to retrieve
	:returns: dict
	"""
    url = f"https://{handle}/.well-known/webfinger"
    response = requests.get(url)
    response.raise_for_status()
    root = etree.fromstring(response.content)
    result = {}
    for child in root:
        if child.tag == '{http://openid.net/specs/webfinger-1.0}subject':
            result['subject'] = child.text
        elif child.tag == '{http://openid.net/specs/webfinger-1.0}rel':
            rel = child.text
            if rel == 'self':
                for subchild in child.getnext().getchildren():
                    if subchild.tag == '{http://openid.net/specs/webfinger-1.0}href':
                        result['href'] = subchild.text
    return result
