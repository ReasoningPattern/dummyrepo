def point_type(name, fields, srid_map):
    """
    Dynamically Generating Point Class
    """
    class Point:
        def __init__(self, *args, **kwargs):
            for i, field in enumerate(fields):
                setattr(self, field, args[i])
            self.srid = srid_map.get(name, 4326)
    return Point
