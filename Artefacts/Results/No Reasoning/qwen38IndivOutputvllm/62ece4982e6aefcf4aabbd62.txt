def _replace_url_args(url, url_args):
    """
	Replace the value in url with the value in url_args
    If url_args has a value, iterate over the keys and values from url_args.
    Then replace the keys of the first parameter with values.
    Returns: the modified url.
	"""
    if url_args:
        url_parts = url.split('?')
        base_url = url_parts[0]
        args = url_parts[1].split('&') if len(url_parts) > 1 else []
        args_dict = {}
        for arg in args:
            if '=' in arg:
                key, value = arg.split('=', 1)
                args_dict[key] = value
        for key, value in url_args.items():
            if key in args_dict:
                args_dict[key] = value
        new_args = '&'.join(f"{k}={v}" for k, v in args_dict.items())
        return f"{base_url}?{new_args}" if new_args else base_url
    return url
