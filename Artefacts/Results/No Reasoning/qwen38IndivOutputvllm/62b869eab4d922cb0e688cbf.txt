def generate_default_observer_schema(app):
    """
    Generate the default observer schema for each Kubernetes resource present in ``spec.manifest`` for which a custom observer schema hasn't been specified.
    """
    default_observer_schema = {
        "type": "object",
        "properties": {
            "resource": {
                "type": "string",
                "description": "The Kubernetes resource name."
            },
            "namespace": {
                "type": "string",
                "description": "The namespace of the Kubernetes resource."
            },
            "uid": {
                "type": "string",
                "description": "The UID of the Kubernetes resource."
            },
            "creation_timestamp": {
                "type": "string",
                "description": "The creation timestamp of the Kubernetes resource."
            },
            "labels": {
                "type": "object",
                "description": "Labels associated with the Kubernetes resource."
            },
            "annotations": {
                "type": "object",
                "description": "Annotations associated with the Kubernetes resource."
            },
            "status": {
                "type": "object",
                "description": "Status information of the Kubernetes resource."
            }
        },
        "required": ["resource", "namespace", "uid", "creation_timestamp"]
    }
    return default_observer_schema
