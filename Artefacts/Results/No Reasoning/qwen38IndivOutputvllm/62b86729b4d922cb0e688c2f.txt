def base_config(user, etcd_host="localhost", etcd_port=2379):
    """
    Creates a configuration with some simple parameters, the key parameters contains "tls", "authentication", "authorization", "etcd", "docs", and "log".
    """
    config = {
        "tls": {
            "enabled": False,
            "cert_file": "/etc/ssl/certs/tls.crt",
            "key_file": "/etc/ssl/private/tls.key"
        },
        "authentication": {
            "enabled": True,
            "type": "token",
            "token": user
        },
        "authorization": {
            "enabled": True,
            "type": "rbac"
        },
        "etcd": {
            "host": etcd_host,
            "port": etcd_port,
            "prefix": "/config"
        },
        "docs": {
            "enabled": True,
            "path": "/usr/share/docs"
        },
        "log": {
            "level": "info",
            "file": "/var/log/app.log"
        }
    }
    return config
