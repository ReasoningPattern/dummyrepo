def load_configurations(config_filenames, overrides=None, resolve_env=True):
    import logging
    from configparser import ConfigParser
    from pathlib import Path

    config_results = []
    error_logs = []

    for filename in config_filenames:
        filename_path = Path(filename)
        if not filename_path.exists():
            error_logs.append(logging.LogRecord(
                logger=logging.getLogger(__name__),
                level=logging.WARNING,
                message=f"Configuration file {filename} does not exist.",
                args={},
                exc_info=False
            ))
            continue

        parser = ConfigParser()
        try:
            parser.read(filename)
        except Exception as e:
            error_logs.append(logging.LogRecord(
                logger=logging.getLogger(__name__),
                level=logging.ERROR,
                message=f"Error reading configuration file {filename}: {str(e)}",
                args={},
                exc_info=False
            ))
            continue

        config = parser._intermediate  # Access the parsed configuration
        config_results.append((filename_path.name, config))

    return config_results, error_logs
