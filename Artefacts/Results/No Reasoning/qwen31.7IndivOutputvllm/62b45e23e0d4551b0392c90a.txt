def validate_version_inventories(self, version_dirs):
    """Each version SHOULD have an inventory up to that point. Also keep a record of any content digests different from those in the root inventory so that we can also check them when validating the content.

    version_dirs is an array of version directory names and is assumed to be in version sequence (1, 2, 3...).
    """
    # Check if version_dirs is not empty
    if not version_dirs:
        raise ValueError("version_dirs cannot be empty")

    # Assume that each version directory exists and is in the correct sequence
    for version in version_dirs:
        # Check if inventory exists for this version
        if not self._check_inventory_exists(version):
            raise ValueError(f"Inventory for version {version} does not exist")

        # Check if content digest is different from root inventory
        if self._check_content_digest_difference(version):
            self._record_content_digest(version)
