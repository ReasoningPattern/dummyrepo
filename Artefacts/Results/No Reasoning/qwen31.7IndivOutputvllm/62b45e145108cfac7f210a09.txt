def check_digests_present_and_used(self, manifest_files, digests_used):
    """Check all digests in manifest that are needed are present and used. Return error() in the class."""
    # Assume manifest_files contains the list of manifests to check
    # Assume digests_used contains the digests that are used
    # For simplicity, check if all required digests are present in the used digests
    # This is a placeholder implementation
    required_digests = set(manifest_files)
    used_digests = set(digests_used)
    if required_digests.issubset(used_digests):
        return None  # No error
    else:
        return self.error("Digests in manifest are not present or used.")
