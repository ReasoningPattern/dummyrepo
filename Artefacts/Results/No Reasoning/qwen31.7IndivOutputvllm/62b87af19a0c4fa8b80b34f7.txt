def difference(d1, d2, level=-1):
	"""
	Return a dictionary with items from d1 not contained in d2.
	"""
	result = {}
	for key, value in d1.items():
		if level == 0:
			if key not in d2:
				result[key] = value
		else:
			if key not in d2:
				result[key] = value
	return result
