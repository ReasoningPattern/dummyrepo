import requests
from xml.etree import ElementTree as ET

def retrieve_and_parse_diaspora_webfinger(handle):
	"""
	Retrieve a and parse a remote Diaspora webfinger document.

:arg handle: Remote handle to retrieve
:returns: dict
	"""
	url = f"https://webfinger.org/remote/{handle}"
	response = requests.get(url)
	
	if response.status_code != 200:
		raise Exception(f"Failed to retrieve webfinger document: {response.status_code}")
	
	root = ET.fromstring(response.content)
	
	# Extract the remote address
	remote_address = root.find('.//remote').get('href')
	
	# Extract the user info
	user_info = {}
	user_info['username'] = root.find('.//username').text
	user_info['email'] = root.find('.//email').text
	user_info['uri'] = remote_address
	
	return user_info
