import re

def _get_resource_name_regex():
    """
    Return the regular expressions that are used to validate the name of the Krake resources
    """
    # Regular expressions for valid Krake resource names
    # Name must be at least 1 character and at most 63 characters
    # Can contain letters, digits, underscores, and hyphens
    # Must not start or end with a hyphen
    # Must not contain consecutive hyphens
    # Must not contain invalid characters (e.g., spaces, parentheses)
    pattern = r'^[a-zA-Z0-9_\-]{1,63}$'
    # Additional checks for invalid characters
    pattern += r'[^a-zA-Z0-9_\- ]'
    return pattern
