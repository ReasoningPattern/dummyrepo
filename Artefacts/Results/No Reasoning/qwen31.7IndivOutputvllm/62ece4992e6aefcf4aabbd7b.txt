def write_configuration(config_filename, rendered_config, mode=0o600, overwrite=False):
	"""
	Given a rendered config YAML, write it out to target file.
    But if the file already exists and overwrite is False,
    abort before writing anything.
    If the file does not exist, create it.
    Write to the file otherwise.

    Returns: None
	"""
	import os
	import yaml

	if overwrite:
		with open(config_filename, 'w') as f:
			yaml.safe_dump(rendered_config, f, default_flow_style=False, encoding='utf-8', allow_unicode=True)
	else:
		if os.path.exists(config_filename):
			print(f"File {config_filename} already exists. Overwrite is False, aborting.")
			return
		with open(config_filename, 'w') as f:
			yaml.safe_dump(rendered_config, f, default_flow_style=False, encoding='utf-8', allow_unicode=True)
	os.chmod(config_filename, mode)
