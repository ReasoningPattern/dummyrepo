def _extract_number_and_supplment_from_issue_element(issue):
	"""
	Return the possible values of number and sup from the contents of issue.
	"""
	# Assuming issue is a string or a dictionary
	if isinstance(issue, str):
		return {'number': None, 'sup': None}
	elif isinstance(issue, dict):
		number = issue.get('number', None)
		sup = issue.get('sup', None)
		return {'number': number, 'sup': sup}
	else:
		return {'number': None, 'sup': None}
