def deep_merge_nodes(nodes):
    result = {}
    for node in reversed(nodes):
        for k, v in node.items():
            if k in result:
                existing = result[k]
                if isinstance(existing, dict) and isinstance(v, dict):
                    merged = deep_merge_nodes([v, existing])
                    result[k].update(merged)
            else:
                result[k] = v
    return result
