from typing import Sequence
from random import Random

FIXED_RANDOM_SEED = 42

def _shuffled(seq: Sequence[str]) -> list[str]:
    """
    Shuffle a given seq with the given FIXED_RANDOM_SEED
    """
    lst = list(seq)
    rng = Random()
    rng.seed(FIXED_RANDOM_SEED)
    rng.shuffle(lst)
    return lst
