def extend_cli(self, root_subparsers):
    """
    Adds the spec cli options to to the main entry point.

    :param subparser: the subparser object to extend.
    """
    parser = root_subparsers.add_parser('spec', help='spec command')
    parser.add_argument('--option', help='spec option description', required=True)
    parser.add_argument('--another-option', type=int, help='another spec option')
    parser.set_defaults(func=self.run_spec)
