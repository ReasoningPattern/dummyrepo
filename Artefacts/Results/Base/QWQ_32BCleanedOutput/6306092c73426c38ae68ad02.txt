def validate_arg_deprecation(self, cli_args, answer_file_args):
    deprecated = set()
    for arg in cli_args:
        if arg in self.deprecated_arguments:
            deprecated.add(arg)
    for arg in answer_file_args:
        if arg in self.deprecated_arguments:
            deprecated.add(arg)
    for arg in deprecated:
        print(f"Warning: The argument '--{arg}' is deprecated.")
