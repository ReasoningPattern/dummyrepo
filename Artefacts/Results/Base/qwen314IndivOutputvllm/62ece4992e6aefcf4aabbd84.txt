import ipaddress

def is_ipv4(target):
    """
    Test if IPv4 address or not.

    Returns: Boolean, True if target is IPv4 address, else False.
    """
    if not isinstance(target, str):
        return False
    try:
        ipaddress.IPv4Address(target)
        return True
    except ipaddress.AddressValueError:
        return False
