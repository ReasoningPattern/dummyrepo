import xml.etree.ElementTree as ET
import xmlsec

def verify_relayable_signature(public_key, doc, signature):
    """
    Verify the signed XML elements to have confidence that the claimed
    author did actually generate this message.
    """
    try:
        # Parse the document and signature XML
        doc_tree = ET.fromstring(doc)
        sig_tree = ET.fromstring(signature)
        
        # Load the public key
        key = xmlsec.Key.load_pem(public_key)
        if not key:
            return False
        
        # Verify the detached signature against the document
        xmlsec.verify_detached(sig_tree, doc.encode(), key)
        return True
    except (ET.ParseError, xmlsec.Error):
        return False
