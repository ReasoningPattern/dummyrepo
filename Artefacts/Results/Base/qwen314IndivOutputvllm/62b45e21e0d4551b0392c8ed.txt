import os

def find_path_type(path):
    """
    Return a string indicating the type of thing at the given path
    """
    if not os.path.exists(path):
        return 'does not exist'
    elif os.path.isfile(path):
        return 'file'
    elif os.path.isdir(path):
        return 'directory'
    else:
        return 'other'
