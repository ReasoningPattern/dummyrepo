import os

def _eval_file(prefix, file_path):
    if not file_path.startswith(prefix):
        return None
    file_type = os.path.splitext(file_path)[1].lower()
    if file_type == '.pdf':
        return {'component_id': prefix, 'file_path': file_path}
    else:
        return {
            'component_id': prefix,
            'file_path': file_path,
            'ftype': file_type,
            'file_path': file_path
        }
