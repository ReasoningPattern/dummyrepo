import inspect

def get_parser_option_specs(self, command_name):
    """
    Gets all the options for the specified command
    
    param command_name: the command name (main, virsh, ospd, etc...)
    return: the list of all command options
    """
    sig = inspect.signature(self.__class__.__method__)
    parameters = sig.parameters
    options = []
    for param in parameters.values():
        if param.kind in (inspect.Parameter.VAR_POSITIONAL, inspect.Parameter.VAR_KEYWORD):
            continue
        if param.kind == inspect.Parameter.POSITIONAL_ONLY and param.default == inspect.Parameter.empty:
            continue
        if param.kind == inspect.Parameter.POSITIONAL_OR_KEYWORD:
            if param.default == inspect.Parameter.empty and param.name.startswith('-'):
                options.append(param.name)
    return sorted(options)
