import argparse

def parse_arguments(*unparsed_arguments):
    """
    Given command-line arguments with which this script was invoked, parse the arguments and return
    them as a dict mapping from subparser name (or "global") to an argparse.Namespace instance.
    """
    parser = argparse.ArgumentParser()
    namespace = parser.parse_args(unparsed_arguments)
    
    result = {}
    for name, value in namespace.__dict__.items():
        if hasattr(namespace, 'subparser_name') and name == namespace.subparser_name:
            result[name] = value
        elif hasattr(namespace, 'parser') and name == namespace.parser.name:
            result[name] = value
    return result
