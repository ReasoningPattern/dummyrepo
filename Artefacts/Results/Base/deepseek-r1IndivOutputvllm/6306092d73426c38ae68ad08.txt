import inspect

def _get_conditionally_required_args(self, command_name, options_spec, args):
    """
    List arguments with ``required_when`` condition matched.
    
    :param command_name: the command name.
    :param options_spec:  the list of command spec options.
    :param args: the received input arguments
    :return: list, list of argument names with matched ``required_when``
        condition
    """
    args_info = inspect.getfullargspec(command_name)
    args_list = args_info.args if hasattr(args_info, 'args') else []
    required_args = []
    
    for arg_name, arg_info in zip(args_list, (arg_info for arg_info in args_info.args_info if hasattr(arg_info, 'required_when'))):
        if arg_info.required_when:
            required_args.append(arg_name)
            
    return required_args
