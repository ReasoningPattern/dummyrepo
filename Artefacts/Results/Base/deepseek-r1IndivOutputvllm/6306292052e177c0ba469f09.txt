import json
import xml.etree.ElementTree as ET

def identify_request(request: RequestType):
    try:
        body = request.json()
        if 'events' in body:
            return True
        return False
    except json.JSONDecodeError:
        pass
    try:
        body = request.read().decode('utf-8')
        tree = ET.parse ET from body
        root = tree.getroot()
        if 'Magic_ENV_TAG' in root.attrib:
            return True
        return False
    except ET.ParseError:
        return False
