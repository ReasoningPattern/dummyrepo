def validate_arg_deprecation(self, cli_args, answer_file_args):
    """
    Validates and prints the deprecated arguments.

    :param cli_args: the dict of arguments from cli
    :param answer_file_args:  the dict of arguments from files
    """
    for key in cli_args:
        if key not in answer_file_args:
            print(f"Warning: Deprecated argument '{key}'. Please use '{answer_file_args[key]}' instead.")
        elif cli_args[key] != answer_file_args[key]:
            print(f"Warning: Deprecated argument '{key}'. Value has changed from '{cli_args[key]}' to '{answer_file_args[key]}'")
