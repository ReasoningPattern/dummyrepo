def normalized(self):
    """
    Normalize all units of time to integers.
    """
    total = 0
    units = [
        ('second', 1e9),
        ('millisecond', 1e6),
        ('microsecond', 1e3),
        ('nanosecond', 1),
        ('hour', 3600e9),
        ('day', 86400e9),
        ('week', 604800e9),
        ('month', 2592000e9),
        ('year', 31536000e9)
    ]
    for unit, factor in units:
        if hasattr(self, unit):
            total += getattr(self, unit) * factor
    return int(total)
