def find_roots(
    graph: "Graph", prop: "URIRef", roots: Optional[Set["Node"]] = None
) -> Set["Node"]:
    if roots is None:
        roots = set()
    all_nodes = set(graph.subjects()) | set(graph.objects())
    for node in all_nodes:
        if not list(graph.triples((node, prop, None))):
            roots.add(node)
    return roots
