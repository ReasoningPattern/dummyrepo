import socket
import ipaddress

def is_local(host):
    hostname = host.split('@')[-1]
    if hostname in ('localhost', '127.0.0.1', '::1'):
        return True
    try:
        ip = ipaddress.ip_address(hostname)
        return ip.is_loopback
    except ValueError:
        pass
    return hostname == socket.gethostname() or hostname == socket.getfqdn()
