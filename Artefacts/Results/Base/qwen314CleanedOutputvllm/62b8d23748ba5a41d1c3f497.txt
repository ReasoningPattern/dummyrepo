def popitem(self):
    """
    Remove and return the `(key, value)` pair least frequently used.
    """
    if not self._cache:
        raise KeyError("Cache is empty")
    min_freq = min(self._freq.values())
    candidates = [k for k, v in self._freq.items() if v == min_freq]
    key = candidates[0]
    value = self._cache.pop(key)
    self._freq.pop(key)
    return (key, value)
