import os

def _group_files_by_xml_filename(source, xmls, files):
    """
    Groups files by xmls and returns data in dict format.
    """
    grouped = {}
    for xml in xmls:
        xml_base = os.path.splitext(xml)[0]
        grouped[xml] = [f for f in files if os.path.splitext(f)[0] == xml_base]
    return grouped
