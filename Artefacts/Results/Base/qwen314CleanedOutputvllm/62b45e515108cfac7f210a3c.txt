import os
import json

def initialize(self):
    """
    Create and initialize a new OCFL storage root.
    """
    storage_path = os.path.join(self.root, 'storage')
    objects_path = os.path.join(storage_path, 'objects')
    os.makedirs(objects_path, exist_ok=True)
    ocfl_json_path = os.path.join(self.root, 'ocfl.json')
    with open(ocfl_json_path, 'w') as f:
        json.dump({
            "version": "1.1",
            "storageLayout": {
                "type": "directory"
            },
            "manifest": {
                "digestAlgorithm": "SHA-256"
            }
        }, f)
