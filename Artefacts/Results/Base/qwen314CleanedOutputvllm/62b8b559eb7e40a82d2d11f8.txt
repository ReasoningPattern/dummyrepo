def minimalBases(classes):
    """
    Returns all classes without subclasses as a list.
    """
    result = []
    for cls in classes:
        has_subclass = False
        for other in classes:
            if other is not cls and issubclass(other, cls):
                has_subclass = True
                break
        if not has_subclass:
            result.append(cls)
    return result
