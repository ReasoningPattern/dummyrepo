def validate_requires_args(self, required_args, args):
    """
    Check if all the required arguments have been provided.

    Args:
        required_args (list): List of argument names that must be provided.
        args (tuple): Tuple of positional arguments.

    Returns:
        bool: True if all required arguments are present, False otherwise.
    """
    for arg in required_args:
        if arg not in args:
            raise ValueError(f"Missing required argument: {arg}")
    return True
