def parse_subparser_arguments(unparsed_arguments, subparsers):
    parsed_results = {}
    remaining_arguments = list(unparsed_arguments)
    
    for subparser_name, subparser in subparsers.items():
        try:
            arguments = argparse.Namespace(*unparsed_arguments)
            args = subparser.parse_args(args=arguments)
            parsed_results[subparser_name] = args
            unparsed_arguments = remaining_arguments[remaining_arguments.index(args) + len(args)]
        except Exception:
            pass
    
    return (parsed_results, remaining_arguments)
