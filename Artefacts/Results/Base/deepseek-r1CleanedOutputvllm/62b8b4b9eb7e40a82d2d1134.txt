def namesAndDescriptions(self, all=False):
    info = None
    for key, value in self.__class__.__dict__.items():
        if hasattr(value, '__doc__'):
            info = {'name': key, 'description': value.__doc__}
            break
    if info is None:
        return None
    if all:
        return info
    else:
        return f"{info['name']}: {info['description']}"
