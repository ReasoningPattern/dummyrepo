from dateutil.parser import parse

def isoparse(self, dt_str):
    """
    Parse an ISO-8601 datetime string into a :class:`datetime.datetime`.

    Uses the flexible parser from dateutil to handle various ISO-8601 formats
    including different separators and optional time zone offsets.
    """
    return parse(dt_str, dayfirst=False, sep=None, timespec='S')
