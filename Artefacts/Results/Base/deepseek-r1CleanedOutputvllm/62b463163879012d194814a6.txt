import zipfile

def files_list_from_zipfile(zip_path):
    """
    Return the files in the given zip path.
    """
    try:
        with zipfile.ZipFile(zip_path, 'r') as f:
            files = []
            for entry in f.listFiles():
                if entry.is_file():
                    files.append(entry.name)
        return files
    except FileNotFoundError:
        raise
