from sphinxext.versioneer import VersioneerConfig

def get_config():
    """
    Return a new VersioneerConfig() and set various attribute of it.
    """
    config = VersioneerConfig()
    
    # Set root URL
    config.root_url = 'https://github.com/'
    
    # Set compiler options
    config.CompilerOptions = {
        'truncation': 200,
        'pretty-prints': True
    }
    
    # Set some configuration options
    config.use_twelve = True
    config.use_abbreviated_month_names = True
    config.force_use_pluralization = True
    config.min_release_version = '2020-01-01'
    
    # Set extensions
    config.extensions = ['sphinxcontrib', 'sphinxext', 'myst']
    
    # Set language configuration
    config.languages = ['en', 'es', 'fr']
    config.default_lang = 'en'
    
    # Set copy options
    config.copy = {
        'url': 'https://github.com/',
        'srcdir': '.',
        'outdir': 'docs'
    }
    
    # Set root namespace
    config.root_namespace = 'https://github.com/'
    
    # Set html options
    config.html = {
        'code-blocks': True,
        'download-links': True,
        'toctree': True
    }
    
    # Set theme
    config.theme = 'sphinx'
    config.html_theme_options = {
        'colors': {
            'bg': '#f0f0f0',
            'darker_text': '#1a1a1a',
            'darker Background': '#1a1a1a'
        }
    }
    
    # Add some additional configuration
    config.html_sidekicks = True
    config.html_distalink = True
    
    # Set custom meta description
    config.html_meta = {
        'description': 'GitHub - Python Software Engineer'
    }
    
    return config
