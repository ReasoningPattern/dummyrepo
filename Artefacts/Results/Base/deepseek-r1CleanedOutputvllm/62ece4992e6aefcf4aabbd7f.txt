from urllib.parse import urlparse

def _parse_image_ref(image_href: str) -> Tuple[str, str, bool]:
    """
    Parse an image href into composite parts import urllib.

    :param image_href: href of an image
    :returns: a tuple of the form (image_id, netloc, use_ssl)
    :raises ValueError:
    """
    parsed_url = urlparse(image_href)
    if not parsed_url:
        raise ValueError("Invalid image href")
    
    netloc = parsed_url.netloc
    if not parsed_url.path:
        raise ValueError("Image reference does not contain an image path")
    
    image_id = parsed_url.path
    use_ssl = parsed_url.scheme == 'https'
    
    return (image_id, netloc, use_ssl)
