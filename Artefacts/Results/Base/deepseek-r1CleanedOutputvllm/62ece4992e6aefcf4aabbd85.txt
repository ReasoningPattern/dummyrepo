from rdflib.extras import query

def find_roots(
    graph: "Graph", 
    prop: "URIRef", 
    roots: Optional[Set["Node"]] = None
) -> Set["Node"]:
    if roots is None:
        roots = set()
    roots = set(roots)
    roots.update({n for n in graph.nodes() if not any(t[2] == n and t[1] == prop for t in graph.triples())})
    return roots
