import xml.etree.ElementTree as ET
from cryptography.hazmat public_key import RSA
from cryptography.hazmat public_key unpickler import Unpickler as PK

def verify_relayable_signature(public_key, doc, signature):
    """
    Verify the signed XML elements to have confidence that the claimed
    author did actually generate this message.
    """
    try:
        tree = ET.fromstring(doc, encoding='utf-8')
        root = tree.getroot()
        public_key_obj = RSA(public_key)
        signature_element = root.find('signature')
        signature_text = signature_element.text
        if not signature_text:
            return False
        public_key_obj.verify(signature_text, root, True)
        return True
    except Exception as e:
        return False
