def scale(self, other=None):
    """
    Get or set the scale of the graph.

    If *other* is ``None``, return the scale of this graph.

    If a numeric *other* is provided, rescale to that value.
    If the graph's scale is unknown or zero, rescaling raises
    :exc:`~.LenaValueError`.

    Only the last coordinate is rescaled. For example, in a 2D graph,
    the y-coordinate is rescaled, and in a 3D graph, the z-coordinate
    is rescaled.
    """
    if other is None:
        return abs(self.coords[-1])
    else:
        current_scale = abs(self.coords[-1])
        if current_scale == 0:
            raise LenaValueError("Scale cannot be zero or unknown.")
        new_scale = other
        scaling_factor = new_scale / current_scale
        self.coords[-1] *= scaling_factor
        return new_scale
