import inspect

def bash_completion():
    """
    Produce the borgmatic command by introspecting borgmatic's command-line argument parsers.
    """
    module = inspect.getmodule(bash_completion)
    sig = inspect.signature(module.__main__)
    args = inspect.getfullargspec(sig).args
    return f"Usage: {module.__name__} {args}\n" + "\n".join(f"  {arg}" for arg in args)
