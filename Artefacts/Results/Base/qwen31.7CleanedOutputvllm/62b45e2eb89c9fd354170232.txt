def next_version(version):
    parts = list(map(int, version.split('.')))
    parts[-1] += 1
    if parts[-1] > 9:
        parts[-1] = 0
        if parts[-2] == 9:
            parts[-2] = 0
            while parts[-1] == 0 and parts[-2] == 9:
                parts[-2] += 1
                parts[-1] = 0
    return '.'.join(map(str, parts))
