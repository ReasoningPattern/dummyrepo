def _normalizeargs(sequence, output=None):
    """
    Normalize declaration arguments.

    Args:
        sequence: A sequence of arguments to normalize.
        output: Optional output list to fill with normalized arguments.

    Returns:
        The normalized sequence.
    """
    if output is None:
        output = []
    
    for item in sequence:
        if isinstance(item, tuple):
            output.append(list(item))
        elif isinstance(item, str):
            # Assume that any string is a single interface or implements spec
            # and expand it if not a single interface
            # This is a simplified check; actual logic may vary
            output.append(list(item))
        else:
            output.append(list(item))
    
    return output
