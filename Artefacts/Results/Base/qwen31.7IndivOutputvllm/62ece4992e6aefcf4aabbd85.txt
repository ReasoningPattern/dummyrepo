

from rdflib import Graph, URIRef

def find_roots(graph, prop, roots=None):
    if roots is None:
        roots = set()
    all_nodes = graph.nodes()
    for node in all_nodes:
        has_parent = False
        query = """
            SELECT ?parent
            WHERE {
                ?parent ?prop ?node.
            }
        """
        results = graph.query(query)
        if results:
            has_parent = True
        if not has_parent:
            roots.add(node)
    return roots
