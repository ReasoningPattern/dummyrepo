def _legacy_mergeOrderings(orderings):
	"""
	Combine multiple lists in the sequence of occurrence into a list with no duplicate elements.
	"""
	flattened = []
	for ordering in orderings:
		flattened.extend(ordering)
	seen = set()
	result = []
	for item in flattened:
		if item not in seen:
			seen.add(item)
			result.append(item)
	return result
