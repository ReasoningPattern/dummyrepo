def deep_merge_nodes(nodes):
    def merge(d1, d2):
        for key, value in d2.items():
            if key in d1:
                if isinstance(d1[key], dict) and isinstance(value, dict):
                    d1[key] = merge(d1[key], value)
                else:
                    d1[key] = value
            else:
                d1[key] = value
        return d1

    result = {}
    for node in nodes:
        result = merge(result, node)
    return result
