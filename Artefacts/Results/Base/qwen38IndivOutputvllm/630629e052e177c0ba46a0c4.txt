import json
from typing import Dict

def parse_diaspora_webfinger(document: str) -> Dict:
    data = json.loads(document)
    hcard_url = None
    if 'links' in data:
        for link in data['links']:
            if 'rel' in link and 'href' in link and link['rel'] == 'hcard':
                hcard_url = link['href']
                break
    return {'hcard_url': hcard_url}
