import argparse

def parse_arguments(*unparsed_arguments):
    """
    Given command-line arguments with which this script was invoked, parse the arguments and return
    them as a dict mapping from subparser name (or "global") to an argparse.Namespace instance.
    """
    parser = argparse.ArgumentParser()
    subparsers = parser.add_subparsers(dest='subparser_name')
    # Example subparsers (this is a simplified version; actual subparsers would be defined elsewhere)
    subparsers.add_parser('cmd1')
    subparsers.add_parser('cmd2')
    parsed = parser.parse_args(unparsed_arguments)
    result = {'global': parsed}
    if parsed.subparser_name:
        result[parsed.subparser_name] = parsed
    return result
