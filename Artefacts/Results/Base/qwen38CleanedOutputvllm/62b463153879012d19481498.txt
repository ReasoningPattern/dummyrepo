import os

def files_list(path):
    """
    Return the files in given path.
    """
    if not os.path.isdir(path):
        return []
    files = []
    for item in os.listdir(path):
        item_path = os.path.join(path, item)
        if os.path.isfile(item_path):
            files.append(item)
    return files
