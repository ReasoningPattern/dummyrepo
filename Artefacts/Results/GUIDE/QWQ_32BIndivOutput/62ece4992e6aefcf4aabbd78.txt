import socket
import ipaddress

def is_local(host):
    """
    Checks if the host is the localhost,
    the localhost include local IP, user name, local domain name, `localhost` and `127.0.0.1`

    Args:
        host: The hostname or ip

    Returns:
        True if the host is the localhost else False
    """
    # Check known loopback names and addresses
    if host in {'localhost', '127.0.0.1', '::1'}:
        return True

    # Check system's hostname and FQDN
    try:
        if host == socket.gethostname() or host == socket.getfqdn():
            return True
    except (socket.error, AttributeError):
        pass  # Handle cases where socket functions might fail

    # Check if it's an IPv4/IPv6 loopback address
    try:
        ip = ipaddress.ip_address(host)
        return ip.is_loopback
    except ValueError:
        pass  # Not a valid IP address

    return False
