def vertex3tuple(vertices):
    """
    Get 3 points for each vertex of the polygon.
    This will include the vertex and the 2 points on both sides of the vertex.
    If the subscript is out of bounds, take the value using modulo to wrap around.
    Args:
        vertices: vertices to be converted

    Returns:
        A list where each element is a tuple of three points
    """
    n = len(vertices)
    if n == 0:
        return []
    result = []
    for i in range(n):
        prev = (i - 1) % n
        next_ = (i + 1) % n
        triplet = (vertices[prev], vertices[i], vertices[next_])
        result.append(triplet)
    return result
