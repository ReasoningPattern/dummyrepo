import re

def process_text_links(text):
    """
    Process links in text, adding some attributes and linkifying textual links.
    """
    # Step 1: Add target and rel attributes to existing <a> tags
    text = re.sub(r'(<a [^>]*>)', r'\1 target="_blank" rel="nofollow"', text)
    # Step 2: Linkify plain text URLs with the required attributes
    text = re.sub(r'((?:https?|ftp)://[^\s<]+)', r'<a href="\1" target="_blank" rel="nofollow">\1</a>', text)
    return text
