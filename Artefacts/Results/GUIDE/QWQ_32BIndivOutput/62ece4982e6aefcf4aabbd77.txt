import datetime

def parse_frequency(frequency):
    """
    Given a frequency string with a number and a unit of time, return a corresponding
    datetime.timedelta instance.
    If the frequency is None or "always", return None.
    Raise ValueError if the given frequency cannot be parsed.
    For instance, given "3 timeunit", return datetime.timedelta(timeunit=3)

    @param frequency: A frequency string "number timeunit"

    @return: datetime.timedelta or None
    """
    if frequency is None or frequency == 'always':
        return None

    parts = frequency.split()
    if len(parts) != 2:
        raise ValueError("Frequency must be in the form 'number timeunit'")

    try:
        number = int(parts[0])
    except ValueError:
        raise ValueError(f"Invalid number: {parts[0]}")

    unit = parts[1]
    allowed_units = {'days', 'seconds', 'microseconds', 'milliseconds', 'weeks'}

    if unit not in allowed_units:
        raise ValueError(f"Invalid unit: {unit}")

    return datetime.timedelta(**{unit: number})
