class VersioneerConfig:
    def __init__(self):
        self.vcs = None
        self.style = None
        self.versionfile_source = None

def get_config():
    """
    Return a new VersioneerConfig() and set various attribute of it.
    """
    config = VersioneerConfig()
    config.vcs = 'git'
    config.style = 'pep440'
    config.versionfile_source = 'setup.py'
    return config
