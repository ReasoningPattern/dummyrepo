import types

def verifyClass(iface, candidate, tentative=False):
    required = []
    for name in iface.__dict__:
        if not name.startswith('_'):
            required.append(name)
    for name in required:
        if not tentative:
            if name not in candidate.__dict__:
                return False
        else:
            if not hasattr(candidate, name):
                return False
    return True
