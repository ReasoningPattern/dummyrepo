class MyClass:
    def __init__(self):
        self.commands = {}

    def get_parser_option_specs(self):
        """Returns a list of all options for all commands."""
        return self.commands.values()

    def get_option_spec(self, command_name, argument_name):
        """Gets the specification for the specified option name.

        Args:
            command_name (str): Name of the command.
            argument_name (str): Name of the argument.

        Returns:
            dict: Specification for the argument in the command.

        Raises:
            ValueError: If the command or argument is not found.
        """
        if command_name not in self.commands:
            raise ValueError(f"Command {command_name} not found")
        if argument_name not in self.commands[command_name]:
            raise ValueError(f"Argument {argument_name} not found in command {command_name}")
        return self.commands[command_name][argument_name]
