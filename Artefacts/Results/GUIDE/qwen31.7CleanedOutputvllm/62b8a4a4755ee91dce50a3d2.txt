from datetime import datetime, timezone
from dateutil import tz

def _fromutc(self, dt):
    """
    Given a timezone datetime in a given timezone, calculates a timezone datetime in a new timezone.
    """
    # Convert dt to UTC
    utc_dt = dt.astimezone(timezone.utc)
    # Convert UTC to new timezone (assumed to be the same as original)
    new_dt = utc_dt.astimezone(tz.gettz(self.timezone))
    return new_dt
