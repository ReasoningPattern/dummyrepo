import functools

def deprecated(message):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            if func.__doc__:
                func.__doc__ += f"\nDeprecated: {message}"
            else:
                func.__doc__ = f"Deprecated: {message}"
            return func(*args, **kwargs)
        return wrapper
    return decorator
