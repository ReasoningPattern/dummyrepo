LEEGL_TAG_CHARS = {'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l', 'm', 'n', 'o', 'p', 'q', 'r', 's', 't', 'u', 'v', 'w', 'x', 'y', 'z'}

def test_tag(tag: str) -> bool:
    allowed = LEEGL_TAG_CHARS
    for char in allowed:
        if char in tag:
            return False
    return True
