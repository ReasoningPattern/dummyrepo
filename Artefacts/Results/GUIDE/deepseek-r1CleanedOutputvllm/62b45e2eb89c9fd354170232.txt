def next_version(version):
    version_parts = list(map(int, version.split('.')))
    carry = 1
    i = len(version_parts) - 1
    while i >= 0 and carry:
        current = version_parts[i] + carry
        if current >= 10:
            version_parts[i] = 0
            carry = 1
        else:
            version_parts[i] = current
            carry = 0
        i -= 1
    if carry:
        version_parts.insert(0, carry)
    return '.'.join(map(str, version_parts))
