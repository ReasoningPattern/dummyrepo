import os
from xml.etree import ElementTree as ET
from pathlib import Path

def _explore_folder(folder):
    """
    Groups files in the given group by using _group_files_by_xml_filename.
    
    This function processes all XML files within the specified folder,
    grouping them and writing their filenames to separate text files.
    """
    for filename in os.listdir(folder):
        file_path = os.path.join(folder, filename)
        if os.path.isfile(file_path) and filename.lower().endswith('.xml'):
            try:
                tree = ET.parse(file_path)
                root = tree.getroot()
                
                # Assuming _group_files_by_xml_filename returns a dictionary
                # where keys are filenames and values are grouped data
                data = _group_files_by_xml_filename(root)
                
                for key, value in data.items():
                    output_path = os.path.join(folder, f"{key}.txt")
                    with open(output_path, 'w') as f:
                        f.write(f"Original Filename: {filename}\n")
                        f.write(f"Grouped Data: {value}\n")
                        f.write(f"Output Filename: {key}.txt\n")
            except Exception as e:
                print(f"Error processing file {filename}: {e}")
