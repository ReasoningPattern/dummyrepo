import inspect

def bash_completion():
    """
    Produce the borgmatic command by introspecting borgmatic's command-line argument parsers.
    """
    argspec = inspect.getfullargspec(bash_completion)
    
    command = "borgmatic "
    for i, name in enumerate(argspec.args):
        if i == 0:
            command += f"-{name}"
        else:
            command += f" -{name}"
    
    return command
