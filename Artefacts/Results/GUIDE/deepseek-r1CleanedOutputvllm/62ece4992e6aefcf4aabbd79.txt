import glob
import os
from six import text_type

def make_find_paths(find_paths):
    def is_glob(path):
        try:
            glob.parse(path)
            return True
        except glob.GlobError:
            return False

    transformed = []
    for path in find_paths:
        if is_glob(path):
            transformed.append(path)
        else:
            transformed.append(text_type(path) + '*')
    return tuple(transformed)
