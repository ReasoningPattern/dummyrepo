class Structure:
    """A simple class to hold structured data from a timedelta."""
    
    def __init__(self, days=0, seconds=0, microseconds=0):
        self.days = days
        self.seconds = seconds
        self.microseconds = microseconds

def dehydrate_timedelta(value):
    """
    Convert a timedelta to a Structure instance.
    
    Args:
        value (datetime.timedelta): The timedelta to convert.
        
    Returns:
        Structure: An instance containing the extracted values.
    """
    return Structure(
        days=value.days,
        seconds=value.seconds,
        microseconds=value.microseconds
    )
