def discard(self, n=-1, qid=-1, dehydration_hooks=None,
                hydration_hooks=None, **handlers):
    """
    Generates a DISCARD message to be packed into the stream.

    :param n: Number of records to discard (default = -1 (all))
    :param qid: Query ID to discard for (default = -1 (last query))
    :param dehydration_hooks: Mapping from types to dehydration functions
    :param hydration_hooks: Mapping from types to hydration functions
    :param handlers: Handler functions to include in the response
    """
    # Create a message dictionary
    message = {
        'PACK_STREAM': {
            'DISCARD': {
                'n': n,
                'qid': qid
            },
            'HANDLERS': handlers
        }
    }

    # Create an infinite generator if n is -1 (all records)
    if n == -1:
        while True:
            yield message
    else:
        # Generate n number of discard messages
        for _ in range(n):
            yield message
