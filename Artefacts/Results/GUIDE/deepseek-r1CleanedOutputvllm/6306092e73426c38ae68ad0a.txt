import unittest
from unittest.mock import Mock

class TestValidateChoicesArgs(unittest.TestCase):
    def test_validate_choices_args_valid(self):
        mock = Mock()
        mock.choices = ['a', 'b', 'c']
        self.assertEqual(mock.validate_choices_args('a'), True)
        self.assertEqual(mock.validate_choices_args('d'), False)
        
    def test_validate_choices_args_invalid(self):
        mock = Mock()
        mock.choices = ['a', 'b', 'c']
        self.assertEqual(mock.validate_choices_args(1), False)
        self.assertEqual(mock.validate_choices_args({'a': 'b'}), False)
        
    def test_validate_choices_args_empty_choices(self):
        mock = Mock()
        mock.choices = []
        self.assertEqual(mock.validate_choices_args('a'), False)
        
    def test_validate_choices_args_none_arg(self):
        mock = Mock()
        mock.choices = ['a', 'b', 'c']
        self.assertEqual(mock.validate_choices_args(None), False)
        
    def test_validate_choices_args_duplicate_choices(self):
        mock = Mock()
        mock.choices = ['a', 'a', 'b']
        self.assertEqual(mock.validate_choices_args('a'), True)
        self.assertEqual(mock.validate_choices_args('b'), True)
        self.assertEqual(mock.validate_choices_args('c'), False)

if __name__ == '__main__':
    unittest.main()
