import collections

class PopItemCache:
    def __init__(self):
        self._order = collections.deque()
        self._data = {}

    def __setitem__(self, key, value):
        self._order.append(key)
        self._data[key] = value

    def popitem(self):
        if not self._order:
            raise KeyError("No items in cache")
        key = self._order.pop()
        value = self._data[key]
        del self._data[key]
        return (key, value)
