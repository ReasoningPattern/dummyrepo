import logging
import requests

def try_retrieve_webfinger_document(handle: str) -> Optional[str]:
    """
    Try to retrieve an RFC7033 webfinger document.
    
    Args:
        handle (str): The handle to use for fetching the document.
        
    Returns:
        Optional[str]: The filename and content of the document if successful, None otherwise.
    """
    logging.info(f"Attempting to retrieve webfinger document for handle: {handle}")
    
    session = requests.Session()
    response = session.get(handle, stream=True)

    if response.status_code == 200:
        # Parse the Content-Disposition header to get the filename
        headers = response.headers
        content_disposition = headers.get('Content-Disposition')
        if content_disposition:
            # Assuming the format is something like: filename*=utf-8''somefile.txt
            filename = content_disposition.split(',')[1].split('}')[0].strip()
            logging.info(f"Successfully retrieved document with filename: {filename}")
            with open(filename, 'wb') as f:
                f.write(response.content)
            return f"Found document: {filename}"
        else:
            logging.warning("No Content-Disposition header found in response")
    else:
        logging.error(f"Failed to retrieve document. Status code: {response.status_code}")
        return None

    logging.info(f"Successfully retrieved webfinger document for handle: {handle}")
    return None
