import os

class LanguageRenderings:
    def __init__(self):
        self._renditions = {}

    def add_rendition(self, lang, file_path):
        """
        Assign the filepath invoke by filepath() in the class to "lang" in _renditions in the class.
        
        Args:
            lang (str): The language code to use as the key in _renditions
            file_path (str): The file path to assign to the language
        """
        self._renditions[lang] = str(file_path)

    def test_add_rendition(self):
        """
        Unit tests for add_rendition method.
        """
        # Setup
        lang rendered = LanguageRenderings()
        
        # Test adding a rendition
        rendered.add_rendition("es", "es.txt")
        
        # Verify the rendition was added
        self.assertEqual(len(rendered._renditions), 1)
        self.assertIn("es", rendered._renditions)
        self.assertEqual(rendered._renditions["es"], "es.txt")
        
        # Test adding an existing rendition
        rendered.add_rendition("es", "es.txt")
        self.assertEqual(rendered._renditions["es"], "es.txt")
        
        # Test adding a non-string path
        with self.assertRaises(TypeError):
            rendered.add_rendition("es", 123)
