import pytest

class IndexHelper:
    def index(self, key):
        """
        Returns the key in the form of int.
        """
        return int(key)

@pytest.fixture
def helper():
    return IndexHelper()

def test_index_return_type(helper):
    """Test that index returns an int."""
    assert isinstance(helper.index(5), int)

def test_index successfully_converts_string(helper):
    """Test that index successfully converts a string to int."""
    assert helper.index("123") == 123

def test_index_fails_with_non_integers(helper):
    """Test that index raises ValueError with non-integer strings."""
    with pytest.raises(ValueError):
        helper.index("abc")
