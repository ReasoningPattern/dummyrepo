def point_type(name, fields, srid_map):
    """
    Dynamically Generating Point Class
    """
    def __init__(self, **kwargs):
        for field in fields:
            setattr(self, field, kwargs[field])
    
    def get_srid(self, field):
        return srid_map[field]
    
    return type(name, (object,), {
        '__init__': __init__,
        'get_srid': get_srid
    })
