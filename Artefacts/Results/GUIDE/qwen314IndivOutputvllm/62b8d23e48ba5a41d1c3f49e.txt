import random

class MyDict:
    def __init__(self):
        self._data = {}
        self._keys = []

    def __choice(self):
        return random.choice(self._keys)

    def popitem(self):
        """
        Find, remove and return a random `(key, value)` pair via __choice in the class
        """
        if not self._keys:
            raise KeyError('popitem(): dictionary is empty')
        key = self.__choice()
        value = self._data.pop(key)
        self._keys.remove(key)
        return key, value
