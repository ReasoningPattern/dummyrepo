def size_to_bytes(size: str) -> int:
    """
    Convert human readable file size to bytes

    Args:
        size: str, a string representing a human readable file size (eg: '500K')
    Returns:
        int: file size in bytes
    """
    if not size:
        return 0

    unit_char = size[-1].upper()
    if unit_char in ['B', 'K', 'M', 'G', 'T']:
        num_str = size[:-1]
    else:
        num_str = size

    try:
        num = float(num_str)
    except ValueError:
        return 0

    if unit_char == 'B':
        multiplier = 1
    elif unit_char == 'K':
        multiplier = 1024
    elif unit_char == 'M':
        multiplier = 1024 ** 2
    elif unit_char == 'G':
        multiplier = 1024 ** 3
    elif unit_char == 'T':
        multiplier = 1024 ** 4
    else:
        multiplier = 1

    return int(num * multiplier)
