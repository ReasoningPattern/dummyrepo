import re
import unittest

class TestValidateValue(unittest.TestCase):
    def test_valid_string(self):
        self.assertTrue(validate_value("valid_string123"))
    
    def test_invalid_string(self):
        self.assertFalse(validate_value("invalid string!"))
        self.assertFalse(validate_value("ab"))
        self.assertFalse(validate_value("a" * 20))
    
    def test_non_string_value(self):
        self.assertFalse(validate_value(123))
        self.assertFalse(validate_value(123.45))
        self.assertFalse(validate_value(True))
        self.assertFalse(validate_value(None))
    
    def test_empty_string(self):
        self.assertFalse(validate_value(""))

def validate_value(value):
    """
    Validate the given value against the corresponding regular expression.
    """
    if not isinstance(value, str):
        return False
    pattern = r'^[a-zA-Z0-9_]{3,19}$'
    return bool(re.fullmatch(pattern, value))
