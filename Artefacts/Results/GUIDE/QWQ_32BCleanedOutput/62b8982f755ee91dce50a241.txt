class TimeDelta:
    def __init__(self, days=0, hours=0, minutes=0, seconds=0):
        self.days = days
        self.hours = hours
        self.minutes = minutes
        self.seconds = seconds

    @property
    def normalized(self):
        """
        Normalize all units of time to integers.
        """
        total = (self.days * 86400) + (self.hours * 3600) + (self.minutes * 60) + self.seconds
        return int(total)
