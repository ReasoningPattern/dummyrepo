class AssetManager:
    def __init__(self):
        self._assets = {}
    
    def filepath(self, path):
        # This method's implementation is assumed to process the path as needed
        # For example, it might return an absolute path or perform validation
        return path  # Default implementation for demonstration
    
    def add_asset(self, basename, file_path):
        """
        Assign the filepath invoke by filepath() in the class to "basename" in _assets in the class.
        """
        self._assets[basename] = self.filepath(file_path)
