def validate_arg_deprecation(self, cli_args, answer_file_args):
    """
    Validates and prints the deprecated arguments.

    :param cli_args: the dict of arguments from cli
    :param answer_file_args:  the dict of arguments from files
    """
    deprecated_args = getattr(self, "deprecated_args", [])
    found_deprecated = set()

    # Check CLI arguments for deprecated keys
    for arg in cli_args:
        if arg in deprecated_args:
            found_deprecated.add(arg)

    # Check answer file arguments for deprecated keys
    for arg in answer_file_args:
        if arg in deprecated_args:
            found_deprecated.add(arg)

    # Print warnings for each deprecated argument found
    for arg in found_deprecated:
        print(f"Warning: The argument '{arg}' is deprecated. Please avoid using it in future.")
