def parse_version(s: str) -> tuple[int, ...]:
    """
    Converts a string concatenated by dot to a tuple consisting of integers.
    
    Args:
        s: A string representing a version number, e.g., "1.2.3".
        
    Returns:
        A tuple of integers representing the version number.
        
    Raises:
        ValueError: If the input string is invalid (e.g., contains non-numeric characters).
    """
    parts = s.split('.')
    return tuple(map(int, parts))
