import datetime

class Time:
    @classmethod
    def from_ticks(cls, ticks, tz=None):
        if ticks < 0 or ticks >= 86400000000000:
            raise ValueError("ticks out of bounds")
        seconds = ticks / 1e9
        dt = datetime.datetime.utcfromtimestamp(seconds)
        if tz is not None:
            dt = dt.replace(tzinfo=tz)
        return cls(dt)
