import os
import yaml

def get_plugin_spec_flatten_dict(plugin_dir):
    """
    Use YAML to read various information in plugin_dir and return the information in dictionary form.
    
    Args:
        plugin_dir (str): Path to the directory containing YAML files.
        
    Returns:
        dict: Merged dictionary containing all information from YAML files in plugin_dir.
    """
    merged = {}
    for filename in os.listdir(plugin_dir):
        if filename.endswith('.yaml') or filename.endswith('.yml'):
            file_path = os.path.join(plugin_dir, filename)
            with open(file_path, 'r') as f:
                data = yaml.safe_load(f)
                merged.update(data)
    return merged
