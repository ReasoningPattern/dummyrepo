import yaml

def _dump_string(obj, dumper=None):
    """Given a python object and serialize it into a YAML stream

    Args:
        obj: python object
        dumper: optional parameter for custom dumper

    Returns:
        YAML stream
    """
    if dumper is None:
        dumper = yaml.default_dumper
    return yaml.dump(obj, dumper=dumper)
