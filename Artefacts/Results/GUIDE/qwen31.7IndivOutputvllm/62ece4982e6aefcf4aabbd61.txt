def _dictsum(dicts):
    """
    Sum values with the same key in the dictionaries. And return a new dictionary.
    
    Args:
        dicts (list of dict): A list of dictionaries to be summed.
        
    Returns:
        dict: A new dictionary with summed values for each key.
    """
    result = {}
    for d in dicts:
        for key, value in d.items():
            if key in result:
                result[key] += value
            else:
                result[key] = value
    return result
