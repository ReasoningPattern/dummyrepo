def select_filenames_by_prefix(prefix, files):
    """
    For each file in files, return all files that match the given prefix.
    
    Args:
        prefix (str): The prefix to match against file names.
        files (list): A list of file names to check.
    
    Returns:
        list: A list of file names that start with the given prefix.
    """
    return [file for file in files if file.startswith(prefix)]
