import random

class MyDict:
    def popitem(self):
        """
        Find, remove and return a random `(key, value)` pair via __choice in the class
        """
        keys = list(self.data.keys())
        if not keys:
            raise KeyError("dictionary is empty")
        key = random.choice(keys)
        value = self.data.pop(key)
        return (key, value)
