from typing import List

def int_to_string(number: int, alphabet: List[str], padding: Optional[int] = None) -> str:
    digits = []
    for d in str(number):
        index = int(d)
        digits.append(alphabet[index])
    s = ''.join(digits)
    if padding is not None:
        s = s.zfill(padding)
    return s
