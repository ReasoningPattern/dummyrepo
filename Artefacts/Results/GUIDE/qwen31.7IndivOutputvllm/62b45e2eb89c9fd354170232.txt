def next_version(version):
    parts = list(map(int, version.split('.')))
    for i in range(len(parts)-1, -1, -1):
        if parts[i] == 9:
            parts[i] = 0
            if i == 0:
                parts[0] += 1
            else:
                parts[i+1] += 1
        else:
            break
    return '.'.join(map(str, parts))
