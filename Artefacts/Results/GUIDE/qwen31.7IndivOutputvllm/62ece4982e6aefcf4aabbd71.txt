def regex_dict(item):
    """
    Convert *.cpp keys to regex keys.
    Given a dict where the keys are all filenames with wildcards,
    convert only the keys into equivalent regexes and leave the values intact.

    Args:
        item: dict to convert

    Returns:
        dict with keys converted to regexes
    """
    result = {}
    for key in item:
        # Replace wildcards with their regex equivalents
        regex_key = key.replace('*', '.*').replace('?', '.?')
        result[regex_key] = item[key]
    return result
