

```python
import requests
import tarfile
import os
from pathlib import Path

def get_repo_archive(url: str, destination_path: Path) -> Path:
    """
    Retrieve and extract a .tar.gz archive from the given URL, ensuring it contains 'desc' files for each package.
    
    Args:
        url: URL of the .tar.gz archive to download.
        destination_path: Path on disk where the archive is extracted.
        
    Returns:
        Path where the archive is extracted to.
    """
    # Download the archive
    response = requests.get(url)
    if response.status_code != 200:
        raise Exception(f"Failed to download archive: {url}")
    
    # Save the archive to the destination path
    archive_path = destination_path.with_suffix('.tar.gz')
    with open(archive_path, 'wb') as f:
        f.write(response.content)
    
    # Extract the archive
    with tarfile.open(archive_path, 'r:gz') as tar:
        tar.extractall(path=destination_path)
    
    return destination_path
```
