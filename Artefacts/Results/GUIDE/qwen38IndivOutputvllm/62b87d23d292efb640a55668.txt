class VersioneerConfig:
    def __init__(self):
        self.version = None
        self.release = False
        self.dirty = False
        self.vcs = None
        self.vcs_path = None


def get_config():
    """
    Return a new VersioneerConfig() and set various attribute of it.
    """
    config = VersioneerConfig()
    config.version = "0.1.0"
    config.release = True
    config.dirty = False
    config.vcs = "git"
    config.vcs_path = ".git"
    return config
