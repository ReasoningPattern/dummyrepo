import shutil

def normalize_cmd(cmd: tuple[str, ...]) -> tuple[str, ...]:
    """
    Complement the full path to exe and return it in its original form
    """
    if not cmd:
        return cmd
    exe = cmd[0]
    full_exe = shutil.which(exe)
    if full_exe is None:
        return cmd  # Return original form if executable not found
    return (full_exe,) + cmd[1:]
