import re
import unittest

class TestValidateKey(unittest.TestCase):
    def test_valid_key(self):
        self.assertTrue(validate_key("abc123"))
    
    def test_invalid_key_starting_with_number(self):
        self.assertFalse(validate_key("1abc"))
    
    def test_key_too_short(self):
        self.assertFalse(validate_key("ab"))
    
    def test_key_too_long(self):
        self.assertFalse(validate_key("abcdefghijk"))  # 11 characters
    
    def test_key_with_special_characters(self):
        self.assertFalse(validate_key("a!b"))
    
    def test_key_with_valid_characters(self):
        self.assertTrue(validate_key("AbcDef123"))
    
    def test_empty_key(self):
        self.assertFalse(validate_key(""))


def validate_key(key):
    """
    Validate the given key against the corresponding regular expression.
    """
    pattern = r'^[A-Za-z][A-Za-z0-9]{2,9}$'
    return bool(re.match(pattern, key))
