import unittest
from datetime import timedelta


class TestDehydrateTimedelta(unittest.TestCase):
    def test_zero_timedelta(self):
        value = timedelta()
        result = dehydrate_timedelta(value)
        self.assertEqual(result, {'days': 0, 'seconds': 0, 'microseconds': 0})

    def test_one_day(self):
        value = timedelta(days=1)
        result = dehydrate_timedelta(value)
        self.assertEqual(result, {'days': 1, 'seconds': 0, 'microseconds': 0})

    def test_one_hour(self):
        value = timedelta(hours=1)
        result = dehydrate_timedelta(value)
        self.assertEqual(result, {'days': 0, 'seconds': 3600, 'microseconds': 0})

    def test_one_hour_one_minute(self):
        value = timedelta(hours=1, minutes=1)
        result = dehydrate_timedelta(value)
        self.assertEqual(result, {'days': 0, 'seconds': 3660, 'microseconds': 0})

    def test_full_components(self):
        value = timedelta(days=2, hours=3, minutes=4, seconds=5, microseconds=6)
        result = dehydrate_timedelta(value)
        self.assertEqual(result, {'days': 2, 'seconds': 3600 * 3 + 4 * 60 + 5, 'microseconds': 6})


def dehydrate_timedelta(value):
    """
    Use the value in timedelta to generate the Structure class.
    """
    return {
        'days': value.days,
        'seconds': value.seconds,
        'microseconds': value.microseconds
    }
