def cached(cache, key=hashkey, lock=None):
    """
    Returns a decorator function that saves the results in the cache
    """
    def decorator(func):
        def wrapper(*args, **kwargs):
            k = key(args, kwargs)
            if lock is not None:
                lock.acquire()
            try:
                if k in cache:
                    return cache[k]
                else:
                    result = func(*args, **kwargs)
                    cache[k] = result
                    return result
            finally:
                if lock is not None:
                    lock.release()
        return wrapper
    return decorator
