def _inline_r_setup(code: str) -> str:
    """
    Some behaviour of R cannot be configured via env variables, but can
    only be configured via R options once R has started. These are set here.
    """
    setup_code = (
        "options(warn = -1)\n"  # Suppress warnings
        "options(digits = 3)\n"  # Set number of digits for output
        "options(scipen = 999)\n"  # Prefer decimal over scientific notation
        "options(continue = 72)\n"  # Set width of line for continuation
        "options(width = 120)"  # Set maximum width of output lines
    )
    return setup_code + "\n" + code
