def on(self, hook):
    """
    Return a decorator function to add a new handler to the "hook" in the registry in the class.
    """
    def decorator(func):
        if not hasattr(self, '_hooks'):
            self._hooks = {}
        if hook not in self._hooks:
            self._hooks[hook] = []
        self._hooks[hook].append(func)
        return func
    return decorator
