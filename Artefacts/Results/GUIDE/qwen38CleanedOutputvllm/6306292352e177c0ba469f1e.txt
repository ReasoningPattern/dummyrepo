import re

def process_text_links(text):
    """
    Process links in text, adding some attributes and linkifying textual links.
    """
    # Regular expression pattern to match URLs starting with http:// or https://
    url_pattern = re.compile(r'(https?://\S+)')
    # Replace each URL with an anchor tag that includes target="_blank" and rel="noopener"
    processed_text = url_pattern.sub(
        lambda m: f'<a href="{m.group(1)}" target="_blank" rel="noopener">{m.group(1)}</a>', text
    )
    return processed_text
