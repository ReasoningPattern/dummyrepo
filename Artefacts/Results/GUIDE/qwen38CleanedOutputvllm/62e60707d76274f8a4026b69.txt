def point_type(name, fields, srid_map):
    """
    Dynamically Generating Point Class
    """
    def __init__(self, **kwargs):
        for field in fields:
            setattr(self, field, kwargs.get(field))
    return type(name, (object,), {
        '__init__': __init__,
        'fields': fields,
        'srid_map': srid_map
    })
