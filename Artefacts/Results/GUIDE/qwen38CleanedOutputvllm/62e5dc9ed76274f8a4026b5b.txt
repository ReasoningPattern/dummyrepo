import warnings

def deprecated(message):
    """
    Return a decorator function for deprecating functions and methods.
    """
    def decorator(func):
        def wrapper(*args, **kwargs):
            warnings.warn(
                f"The function {func.__name__} is deprecated: {message}",
                DeprecationWarning,
                stacklevel=2
            )
            return func(*args, **kwargs)
        return wrapper
    return decorator
