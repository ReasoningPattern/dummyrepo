import socket

def is_local(host):
    """
    Checks if the host is the localhost,
    the localhost include local IP, user name, local domain name, `localhost` and `127.0.0.1`

    Args:
        host: The hostname or ip

    Returns:
        True if the host is the localhost else False
    """
    # Check for standard localhost representations
    if host in ('localhost', '127.0.0.1'):
        return True

    # Check if host matches the local machine's hostname
    local_hostname = socket.gethostname()
    if host == local_hostname:
        return True

    # Check if host matches the local machine's FQDN
    try:
        local_fqdn = socket.getfqdn()
        if host == local_fqdn:
            return True
    except:
        pass

    # Check if host is a local IP address (excluding loopback)
    try:
        local_ips = set()
        for interface in socket.getaddrinfo(socket.gethostname(), None):
            if interface[0] == socket.AF_INET:
                ip = interface[4][0]
                if ip != '127.0.0.1':
                    local_ips.add(ip)
        if host in local_ips:
            return True
    except:
        pass

    return False
