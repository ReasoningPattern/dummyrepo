def get(self, key, default=None):
    """
    If a key exists in the class, the value corresponding to the key is returned. Otherwise, default is returned.
    """
    return getattr(self, key, default)

class TestClass:
    def __init__(self):
        self.data = {'apple': 1, 'banana': 2, 'cherry': 3}

@pytest.fixture
def obj():
    return TestClass()

@pytest.fixture
def key(request):
    return request.param

def test_get_with_key(obj, key):
    assert obj.get(key, 0) == 2

def test_get_without_key(obj, key):
    assert obj.get(key) == 0

def test_get_with_default(obj, key):
    assert obj.get(key, 100) == 100

if __name__ == "__main__":
    import pytest
    pytest.main()
