import json

def parse_diaspora_webfinger(document: str) -> dict:
    """
    Parses the provided JSON document and extracts the href value from the hcard_url in webfinger.
    
    Args:
        document (str): A JSON formatted string representing the Diaspora webfinger data.
        
    Returns:
        dict: A dictionary containing the extracted href value.
    """
    data = json.loads(document)
    webfinger = data.get('webfinger')
    if webfinger and webfinger.get('hcard_url'):
        url = webfinger['hcard_url']
        return url.get('href')
    return None
