import argparse

def extend_cli(self, subparser):
    """
    Adds the spec cli options to to the main entry point.

    This method extends the subparser with additional options and functionality,
    including help text and version checking.

    :param subparser: The subparser object to extend with new options.
    """
    # Add help text to the subparser
    subparser.add_argument('--help', action='store',
                            dest='help_text',
                            help="Display this help message and exit.")

    # Add version check decorator
    subparser.add decorator(
        @argparse.SUBPARSER_VERSION,
        min_version='2.0'
    )

    # Add common CLI options
    subparser.add_argument('--version',
                           action='version',
                           version='%(prog)s v%(version)s',
                           help='Display the version of the CLI tool.')

    subparser.add_argument('--help',
                           action='store',
                           dest='help_options',
                           help='Display usage help message and exit.')

    # Add option that requires another option to be provided
    subparser.add_argument('--required',
                           action='store_true',
                           dest='required_option',
                           default=False,
                           help='Enforce that this option must be provided.')

    # Add default value for required option
    subparser.add_argument('--default',
                           action='store',
                           dest='default_value',
                           help='Default value if no argument is provided.')

    # Execute the parsing
    subparser.handle()
