def _extract_number_and_supplement_from_issue_element(issue):
    """
    Extract the number and supplementary information from the issue element.
    
    Args:
        issue (str): The content of the issue element.
        
    Returns:
        tuple: A tuple containing the extracted number (int or None) and supplementary information (dict or None).
    """
    # Extract number by checking for numeric values in the issue content
    number = None
    if isinstance(issue, str) and "number=" in issue:
        number = int(issue.split("number=")[1].split(" ")[0])
    
    # Extract supplementary information by checking for common tags or attributes
    supplement = None
    if isinstance(issue, str) and "supplement=" in issue:
        supplement = {
            "info": issue.split("supplement=')[1].split(" ")[0]
        }
    
    return (number, supplement)
