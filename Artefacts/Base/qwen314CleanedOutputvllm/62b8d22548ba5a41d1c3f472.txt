def cachedmethod(cache, key=hashkey, lock=None):
    """
    Returns a decorator function that can call a result in the cache.
    """
    def decorator(func):
        def wrapper(self, *args, **kwargs):
            k = key(self, *args, **kwargs)
            if lock is not None:
                with lock:
                    if k in cache:
                        return cache[k]
                    result = func(self, *args, **kwargs)
                    cache[k] = result
                    return result
            else:
                if k in cache:
                    return cache[k]
                result = func(self, *args, **kwargs)
                cache[k] = result
                return result
        return wrapper
    return decorator
