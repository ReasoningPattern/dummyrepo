import os
import shutil

def normalize_cmd(cmd: tuple[str, ...]) -> tuple[str, ...]:
    """
    Complement the full path to exe and return it in its original form
    """
    if not cmd:
        return cmd
    executable = cmd[0]
    if os.path.isabs(executable):
        return cmd
    full_path = shutil.which(executable)
    if full_path:
        return (full_path,) + cmd[1:]
    return cmd
