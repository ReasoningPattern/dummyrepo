import subprocess

def addignored(ignored):
    result = subprocess.run(['git', 'status', '--porcelain', '--ignored'], capture_output=True, text=True)
    lines = result.stdout.splitlines()
    files = []
    for line in lines:
        if line.startswith('!! '):
            files.append(line[3:].strip())
    files.sort()
    return ','.join(files)
