def _convert_non_cli_args(self, parser_name, values_dict):
    type_map = self._argument_type_map.get(parser_name, {})
    for arg in values_dict:
        if arg in type_map:
            try:
                values_dict[arg] = type_map[arg](values_dict[arg])
            except (ValueError, TypeError):
                pass
