from collections.abc import Sequence

def _get_seq_with_type(seq, bufsize=None):
    if isinstance(seq, Sequence):
        return (seq, type(seq))
    try:
        s = []
        for i in seq:
            s.append(i)
            if bufsize is not None and len(s) >= bufsize:
                break
        return (s, list)
    except TypeError:
        return ([seq], list)
