from pathlib import PurePath

def strip_root(path, root):
    path_obj = PurePath(path)
    root_obj = PurePath(root)
    if path_obj == root_obj:
        return ''
    try:
        return str(path_obj.relative_to(root_obj))
    except ValueError:
        raise ValueError("Path is not under root")
