def normalized(self):
    total_seconds = (
        self.days * 24 * 3600 +
        self.hours * 3600 +
        self.minutes * 60 +
        self.seconds
    )
    total_seconds = int(total_seconds)
    days = total_seconds // (24 * 3600)
    rem = total_seconds % (24 * 3600)
    hours = rem // 3600
    rem %= 3600
    minutes = rem // 60
    seconds = rem % 60
    return self.__class__(days, hours, minutes, seconds)
