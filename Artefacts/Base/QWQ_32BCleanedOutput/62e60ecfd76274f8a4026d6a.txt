def protocol_handlers(cls, protocol_version=None):
    handler_map = {
        1: cls.Bolt1Handler,
        2: cls.Bolt2Handler,
        3: cls.Bolt3Handler,
        4: cls.Bolt4Handler,
    }
    if protocol_version is None:
        return list(handler_map.values())
    return handler_map.get(protocol_version)
