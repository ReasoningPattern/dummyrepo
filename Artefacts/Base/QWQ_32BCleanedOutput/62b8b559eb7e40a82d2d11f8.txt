def minimalBases(classes):
    """
    Returns all classes without subclasses as a list.
    """
    result = []
    for c in classes:
        has_subclass = False
        for d in classes:
            if d is not c and issubclass(d, c):
                has_subclass = True
                break
        if not has_subclass:
            result.append(c)
    return result
