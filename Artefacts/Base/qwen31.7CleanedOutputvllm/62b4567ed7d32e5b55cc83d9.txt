def deep_merge_nodes(nodes):
    result = {}
    for node in nodes:
        if isinstance(node, dict):
            for key, value in node.items():
                if key in result:
                    if isinstance(value, dict):
                        result[key] = deep_merge_nodes([result[key], value])
                    else:
                        result[key] = value
                else:
                    result[key] = value
        else:
            # If the node is not a dict, add it as is, but only if it's the last occurrence
            # This is a simplification; in practice, this would require tracking the last occurrence
            # For the purpose of this function, we assume that non-dict nodes are not part of the key-value pairs
            pass
    return result
