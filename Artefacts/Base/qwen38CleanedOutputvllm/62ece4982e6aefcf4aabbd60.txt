def size_to_bytes(size: str) -> int:
    """
    Convert human readable file size to bytes

    Args:
            size: str, a string representing a human readable file size (eg: '500K')
    Returns:
            int: file size in bytes
	"""
    unit_map = {
        'K': 1024,
        'M': 1024**2,
        'G': 1024**3,
        'T': 1024**4,
        'P': 1024**5,
        'E': 1024**6
    }
    if not size:
        return 0
    unit = size[-1].upper()
    if unit in unit_map:
        num_str = size[:-1]
        num = float(num_str)
        return int(num * unit_map[unit])
    else:
        return int(size)
