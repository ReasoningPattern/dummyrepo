import datetime

def hydrate_time(nanoseconds, tz=None):
    """
    Convert nanoseconds to a time in fixed format.
    """
    seconds = nanoseconds / 1e9
    if tz is not None:
        dt = datetime.datetime.fromtimestamp(seconds, tz)
    else:
        dt = datetime.datetime.utcfromtimestamp(seconds)
    return dt.isoformat()
