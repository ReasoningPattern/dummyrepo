from typing import Tuple, Set
import re

def find_tags(text: str, replacer: callable = None) -> Tuple[Set, str]:
    """
    Find tags in text, ignoring tags inside code blocks.
    If a replacer is provided, replace the tag word with the result of the replacer function called with the tag word.
    Returns a set of tags and the original or replaced text.
    """
    # Find all tags in the text
    tags = set()
    # Use a regular expression to find tags (assuming tags are words separated by spaces)
    # This is a simplified approach and may not handle all cases
    words = re.findall(r'\b\w+\b', text)
    tags = set(words)
    
    # Replace tags with replacer if provided
    if replacer:
        # Create a list to hold the replaced text
        replaced_text = []
        for word in words:
            if word in tags:
                replaced_text.append(replacer(word))
            else:
                replaced_text.append(word)
        replaced_text = ''.join(replaced_text)
    
    return tags, replaced_text
