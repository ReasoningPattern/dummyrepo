def cached(cache, key, lock=None):
    def decorator(func):
        def wrapper(*args, **kwargs):
            key_val = key(*args, **kwargs)
            if key_val in cache:
                return cache[key_val]
            result = func(*args, **kwargs)
            cache[key_val] = result
            if lock is not None:
                lock.acquire()
            return result
        return wrapper
    return decorator
