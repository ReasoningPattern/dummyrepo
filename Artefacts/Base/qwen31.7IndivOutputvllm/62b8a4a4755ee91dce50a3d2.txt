import zoneinfo

def _fromutc(self, dt):
    """Given a timezone datetime in a given timezone, calculates a timezone datetime in a new timezone."""
    # Convert the input datetime to UTC
    utc_dt = zoneinfo.utcfromtimestamp(dt.timestamp())
    # Assume self.original_timezone is the original timezone
    # Convert UTC to the new timezone
    new_dt = zoneinfo.localize(utc_dt, zoneinfo.timezone(self.original_timezone))
    return new_dt
