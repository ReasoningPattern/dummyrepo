def from_raw_values(cls, values):
    """
    Create a Bookmarks object from a list of raw bookmark string values.
    """
    obj = cls()
    for value in values:
        url = value.split()[0]
        title = ' '.join(value.split()[1:])
        setattr(obj, 'url', url)
        setattr(obj, 'title', title)
    return obj
