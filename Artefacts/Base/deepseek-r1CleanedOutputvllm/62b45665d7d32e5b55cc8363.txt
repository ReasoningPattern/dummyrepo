import argparse

def make_parsers():
    """
    Build a parser and its subparsers and return them as a tuple.
    """
    main_parser = argparse.ArgumentParser()
    main_parser.set_description("Main parser description")

    subparsers = {
        "parser1": argparse.ArgumentParser(),
        "parser2": argparse.ArgumentParser(),
    }

    for name, parser in subparsers.items():
        parser.set_description(f"Description for {name}")
        parser.add_argument("argument", action="store")

    return main_parser, subparsers
