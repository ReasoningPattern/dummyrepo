from urllib.parse import urlparse, urlunparse

def _replace_url_args(url, url_args):
    """
    Replace the value in url with the value in url_args
    If url_args has a value, iterate over the keys and values from url_args.
    Then replace the keys of the first parameter with values.
    Returns: the modified url.
    """
    parsed_url = urlparse(url)
    query = parsed_url.query

    if isinstance(url_args, dict):
        parsed_args = urlsplit(url_args)
        query_args = parsed_args.query
        combined_query = parsed_url.query[:-1] if query else '' + url_args.get(
            '', parsed_args.query
        )
        query = combined_query

    new_parsed_url = parsed_url._replace(query=query)
    return urlunparse(new_parsed_url)
