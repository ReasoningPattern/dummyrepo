import argparse

def make_parsers():
    """
    Build a top-level parser and its subparsers and return them as a tuple.
    """
    top_parser = argparse.ArgumentParser(description="A top-level parser that builds subparsers.")
    
    # Get subparsers
    subparsers_result = top_parser.get_subparsers()
    
    # Create dictionary of subparsers
    subparsers = {}
    for subparser_name, subparser in subparsers_result:
        subparsers[subparser_name] = subparser
    
    return top_parser, subparsers
