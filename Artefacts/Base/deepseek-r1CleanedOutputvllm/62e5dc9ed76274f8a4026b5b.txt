def deprecated(message):
    """
    Return a decorator function for deprecating functions and methods.
    """
    def decorator(func):
        if not hasattr(func, 'deprecated'):
            func.deprecated = True
            wrapped_func = None
            def wrapper(*args, **kwargs):
                func.deprecated = True
                raise DeprecationWarning(f"Deprecation warning: {func.__class__.__name__} is deprecated")
            wrapped_func = wrapper
        return wrapped_func
    return decorator
