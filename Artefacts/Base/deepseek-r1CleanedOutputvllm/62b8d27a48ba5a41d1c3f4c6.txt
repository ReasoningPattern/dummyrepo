def cached(cache, key=hashkey, lock=None):
    def decorator(func):
        def wrapper(*args, **kwargs):
            key = key(args, kwargs)
            if lock is not None:
                if lock.is_acquired() != lock.getstate():
                    continue
            result = func(*args, **kwargs)
            cache[key] = result
            return result
        return wrapper
    return decorator
