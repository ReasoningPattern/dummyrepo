import subprocess

def run_command(commands, args, cwd=None, verbose=False, hide_stderr=False, env=None):
    """
    A list of command is given. Run the given command(s) with the given arguments using a subprocess. Return the running result (stdout and returncode).
    """
    results = []
    for command in commands:
        if isinstance(command, str):
            cmd = command
        else:
            cmd = ' '.join([__file__, *command])
        
        proc = subprocess.run(
            cmd,
            args=args,
            cwd=cwd,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            text=True,
            check=False
        )
        
        stdout = proc.stdout.read().decode()
        stderr = proc.stderr.read().decode()
        
        if verbose:
            print(f"Running command: {cmd}")
        
        results.append((stdout, stderr))
    
    return results
