from collections.abc import Iterable

def is_fill_request_seq(seq):
    """
    Check whether seq can be converted to FillRequestSeq and bool is returned.
    """
    if not isinstance(seq, Iterable):
        return False
    if seq is None:
        return False
    for item in seq:
        if not hasattr(item, 'id') or not hasattr(item, 'code') or not hasattr(item, 'description') or not hasattr(item, 'value'):
            return False
    return True
