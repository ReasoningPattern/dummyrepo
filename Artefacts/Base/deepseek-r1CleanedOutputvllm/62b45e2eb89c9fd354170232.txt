def next_version(version):
    parts = list(map(int, version.split('.')))
    if len(parts) < 3:
        parts = [0] * (3 - len(parts)) + parts
    parts[-1] += 1
    i = len(parts) - 1
    while i >= 0 and parts[i] >= 10:
        parts[i] -= 10
        if i == 0:
            parts.insert(0, 1)
            parts.pop()
        else:
            parts[i - 1] += 1
        i -= 1
    return '.'.join(map(str, parts))
