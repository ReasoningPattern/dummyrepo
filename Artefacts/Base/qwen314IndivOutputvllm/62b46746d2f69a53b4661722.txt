import re

def absorb(self, args):
    """
    Given an `args` sequence of expressions, return a new list of expression applying absorption.
    """
    result = []
    for expr in args:
        expr = re.sub(r'(\w+) or \(\1 and (\w+)\)', r'\1', expr)
        expr = re.sub(r'(\w+) and \(\1 or (\w+)\)', r'\1', expr)
        result.append(expr)
    return result
