import os
from collections import defaultdict

def _group_files_by_xml_filename(files):
    groups = defaultdict(list)
    for file in files:
        base, ext = os.path.splitext(file)
        if ext.lower() == '.xml':
            xml_filename = file
            groups[xml_filename].append(file)
        else:
            xml_filename = base + '.xml'
            if xml_filename in groups:
                groups[xml_filename].append(file)
    return groups

def _explore_folder(folder):
    """
    Groups files in the given group by using _group_files_by_xml_filename.
    """
    files = [f for f in os.listdir(folder) if os.path.isfile(os.path.join(folder, f))]
    return _group_files_by_xml_filename(files)
