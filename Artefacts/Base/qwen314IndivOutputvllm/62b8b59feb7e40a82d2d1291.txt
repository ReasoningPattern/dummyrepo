def _getTargetClass(self):
    """
    Define this to return the implementation in use,
    without the 'Py' or 'Fallback' suffix.
    """
    name = self.__class__.__name__
    if name.endswith('Py'):
        return name[:-2]
    elif name.endswith('Fallback'):
        return name[:-8]
    return name
