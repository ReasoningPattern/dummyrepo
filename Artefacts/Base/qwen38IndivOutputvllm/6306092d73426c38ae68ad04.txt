def get_parser_option_specs(self, command_name):
    """
    Gets all the options for the specified command

    :param command_name: the command name (main, virsh, ospd, etc...)
    :return: the list of all command options
    """
    command_options = {
        'main': [
            {'name': '--verbose', 'type': bool, 'help': 'Enable verbose mode'},
            {'name': '--output', 'type': str, 'help': 'Output format'},
        ],
        'virsh': [
            {'name': '--connect', 'type': str, 'help': 'Connect to a specific hypervisor'},
            {'name': '--help', 'type': None, 'help': 'Show help message'},
        ],
        'ospd': [
            {'name': '--config', 'type': str, 'help': 'Path to configuration file'},
            {'name': '--debug', 'type': bool, 'help': 'Enable debug mode'},
        ],
    }
    return command_options.get(command_name, [])
