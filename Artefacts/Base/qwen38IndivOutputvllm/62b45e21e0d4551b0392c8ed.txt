import os

def find_path_type(path):
    """
    Return a string indicating the type of thing at the given path
    """
    if not os.path.exists(path):
        return "not found"
    if os.path.islink(path):
        return "symlink"
    if os.path.isdir(path):
        return "directory"
    if os.path.isfile(path):
        return "file"
    return "unknown"
