from typing import Any
from fastapi import JSONRequest
import json

def identify_request(request: JSONRequest) -> bool:
    """
    Check whether the request body loaded using JSON contains events. If yes, True is returned. Otherwise, False is returned.
    """
    try:
        events = json.loads(request.body)
        return isinstance(events, dict)
    except json.JSONDecodeError:
        return False
