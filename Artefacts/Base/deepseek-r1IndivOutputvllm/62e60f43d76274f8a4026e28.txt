from datetime import datetime

def hydrate_time(nanoseconds, tz=None):
    """
    Convert nanoseconds to a time in fixed format.
    """
    seconds = nanoseconds / 1e9
    dt = datetime.fromtimestamp(seconds)
    if tz is not None:
        dt = dt.astimezone(tz)
    return dt.strftime("%Y-%m-%d %H:%M:%S")
