import re

def _extract_number_and_supplement_from_issue_element(issue):
    number = 0
    supplement = ""
    lines = issue.split('\n')
    
    for line in lines:
        if any(c.isdigit() for c in line):
            number_str = ""
            for c in line:
                if c.isdigit() and (not number_str or not c.isdigit()):
                    number_str += c
            number = int(number_str) if number_str else number
        else:
            url_match = re.search(r'http[s]?://(?:[a-zA-Z]|[0-9]|\.]+\.[a-zA-Z]{2,}', line)
            if url_match:
                supplement += url_match.group(0)
                
    return number, supplement
