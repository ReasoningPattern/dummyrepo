from ansible.module_utils played_outcome import PlayedOutcome
from ansible.playbook.playbook_utils import Playbook
from ansible.utils.ansible_module_utils._util import AnsibleV2Module
from infrared import InfraredWorkspace, InfraredPlugin


def _run_playbook(cli_args, vars_dict, ir_workspace, ir_plugin):
    """
    Runs ansible cli with vars dict

    :param vars_dict: dict, Will be passed as Ansible extra-vars
    :param cli_args: the list  of command line arguments
    :param ir_workspace: An Infrared Workspace object represents the active
                         workspace
    :param ir_plugin: An InfraredPlugin object of the current plugin
    :return: ansible results
    """
    extra_vars = AnsibleV2Module().from_dict(vars_dict)
    playbook = Playbook(module_name="ir_plugin_module", extra_vars=extra_vars, workspace=ir_workspace)
    result = playbook.run(cli_args)
    return result
