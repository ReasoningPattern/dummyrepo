def protocol_handlers(cls, protocol_version=None):
    """
    Different forms of Bolt protocol handlers are returned based on the value of protocol_version.
    """
    handlers = {
        "1.0": cls,  # HTTP handler
        "2.0": cls,  # HTTPS handler
        "3.0": cls,  # WebSocket handler
    }
    if protocol_version is not None:
        return handlers.get(protocol_version, None)
    else:
        return sorted(handlers.values())
