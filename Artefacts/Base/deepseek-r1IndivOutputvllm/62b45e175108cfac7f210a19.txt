def validate_fixity(self, fixity, manifest_files):
    """
    Validate fixity block in inventory. Check the structure of the fixity block and makes sure that only files listed in the manifest are referenced. Return error() in the class.
    """
    errors = []
    for entry in fixity:
        path = entry.path
        if path not in manifest_files:
            errors.append(f"Invalid path: {path}")
        if not entry.is_valid:
            errors.append(f"Invalid entry: {entry}")
    if errors:
        self.error(f"ValidateFixityError: {', '.join(errors)}")
